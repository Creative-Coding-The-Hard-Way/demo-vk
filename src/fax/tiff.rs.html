1#[derive(Copy, Clone)]
2enum Value {
3    Short(u16),
4    Long(u32),
5    Rational(u32, u32),
6    DataOffset,
7}
8
9pub fn wrap(data: &[u8], width: u32, height: u32) -> Vec<u8> {
10    use Value::*;
11    let header_data = [
12        (256, Long(width)), // ImageWidth
13        (257, Long(height)), // ImageLength
14        (259, Short(4)), // Compression
15        (262, Short(0)), // PhotometricInterpretation
16        (273, DataOffset), // StripOffsets
17        (274, Short(1)), // Orientation
18        (278, Long(height)), // RowsPerStrip
19        (279, Long(data.len() as u32)), // StripByteCounts
20        (282, Rational(200, 1)), // XResolution
21        (283, Rational(200, 1)), // YResolution
22        (296, Short(2)), // ResolutionUnit
23    ];
24    let rat_data_len = 2 * 8; // number of rationals * 8
25    let ifd_end = 
26        4 + // magic
27        4 + // IFD offset
28        2 + // IFD entry count
29        12 * header_data.len() + // IFD enties
30        4; // null pointer at end of IFD
31    let header_size = ifd_end + rat_data_len;
32    
33    let mut out = Vec::with_capacity(header_size + data.len());
34
35    out.extend_from_slice(&[73, 73, 42, 0]);
36    let ifd_offset: u32 = 8;
37    out.extend_from_slice(&ifd_offset.to_le_bytes());
38
39    out.extend_from_slice(&u16::to_le_bytes(header_data.len() as u16));
40
41    let mut num_rat = 0;
42    for &(tag, val) in header_data.iter() {
43        let (typ_num, val) = match val {
44            Short(n) => (3, n as u32),
45            Long(n) => (4, n),
46            Rational(_, _) => {
47                let o = ifd_end + 8 * num_rat;
48                num_rat += 1;
49                (5, o as u32)
50            }
51            DataOffset => (4, header_size as u32)
52        };
53        let count = 1;
54        out.extend_from_slice(&u16::to_le_bytes(tag));
55        out.extend_from_slice(&u16::to_le_bytes(typ_num));
56        out.extend_from_slice(&u32::to_le_bytes(count));
57        out.extend_from_slice(&u32::to_le_bytes(val));
58    }
59    // NULL at IFD end
60    out.extend_from_slice(&[0; 4]);
61
62    // write additional data
63    for &(_, val) in header_data.iter() {
64        if let Value::Rational(nom, denom) = val {
65            out.extend_from_slice(&nom.to_le_bytes());
66            out.extend_from_slice(&denom.to_le_bytes());
67        }
68    }
69
70    assert_eq!(out.len(), header_size);
71    out.extend_from_slice(data);
72    out
73}
