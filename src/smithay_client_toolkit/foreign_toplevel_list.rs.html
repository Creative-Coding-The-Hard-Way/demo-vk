8/// Information about a toplevel.
9#[derive(Clone, Debug, Default)]
10#[non_exhaustive]
11pub struct ForeignToplevelInfo {
12    /// Title
13    pub title: String,
14    /// App id
15    pub app_id: String,
16    /// Identifier to check if two toplevel handles refer to same toplevel
17    pub identifier: String,
18}
19
20#[derive(Debug, Default)]
21struct ForeignToplevelInner {
22    current_info: Option<ForeignToplevelInfo>,
23    pending_info: ForeignToplevelInfo,
24}

26#[doc(hidden)]
27#[derive(Debug, Default, Clone)]
28pub struct ForeignToplevelData(Arc<Mutex<ForeignToplevelInner>>);

30#[derive(Debug)]
31pub struct ForeignToplevelList {
32    foreign_toplevel_list: GlobalProxy<ext_foreign_toplevel_list_v1::ExtForeignToplevelListV1>,
33    toplevels: Vec<ext_foreign_toplevel_handle_v1::ExtForeignToplevelHandleV1>,
34}
35
36impl ForeignToplevelList {
37    pub fn new<D>(globals: &GlobalList, qh: &QueueHandle<D>) -> Self
38    where
39        D: Dispatch<ext_foreign_toplevel_list_v1::ExtForeignToplevelListV1, GlobalData> + 'static,
40    {
41        let foreign_toplevel_list = GlobalProxy::from(globals.bind(qh, 1..=1, GlobalData));
42        Self { foreign_toplevel_list, toplevels: Vec::new() }
43    }
44
45    /// Returns list of toplevels.
46    pub fn toplevels(&self) -> &[ext_foreign_toplevel_handle_v1::ExtForeignToplevelHandleV1] {
47        &self.toplevels
48    }
49
50    /// Returns information about a toplevel.
51    ///
52    /// This may be none if the toplevel has been destroyed or the compositor has not sent
53    /// information about the toplevel yet.
54    pub fn info(
55        &self,
56        toplevel: &ext_foreign_toplevel_handle_v1::ExtForeignToplevelHandleV1,
57    ) -> Option<ForeignToplevelInfo> {
58        toplevel.data::<ForeignToplevelData>()?.0.lock().unwrap().current_info.clone()
59    }
60
61    pub fn stop(&self) {
62        if let Ok(toplevel_list) = self.foreign_toplevel_list.get() {
63            toplevel_list.stop();
64        }
65    }
66}

99impl<D> Dispatch<ext_foreign_toplevel_list_v1::ExtForeignToplevelListV1, GlobalData, D>
100    for ForeignToplevelList
101where
102    D: Dispatch<ext_foreign_toplevel_list_v1::ExtForeignToplevelListV1, GlobalData>
103        + Dispatch<ext_foreign_toplevel_handle_v1::ExtForeignToplevelHandleV1, ForeignToplevelData>
104        + ForeignToplevelListHandler
105        + 'static,
106{
107    fn event(
108        state: &mut D,
109        proxy: &ext_foreign_toplevel_list_v1::ExtForeignToplevelListV1,
110        event: ext_foreign_toplevel_list_v1::Event,
111        _: &GlobalData,
112        conn: &Connection,
113        qh: &QueueHandle<D>,
114    ) {
115        match event {
116            ext_foreign_toplevel_list_v1::Event::Toplevel { toplevel: _ } => {}
117            ext_foreign_toplevel_list_v1::Event::Finished => {
118                state.finished(conn, qh);
119                proxy.destroy();
120            }
121            _ => unreachable!(),
122        }
123    }
124
125    wayland_client::event_created_child!(D, ext_foreign_toplevel_list_v1::ExtForeignToplevelListV1, [
126        ext_foreign_toplevel_list_v1::EVT_TOPLEVEL_OPCODE => (ext_foreign_toplevel_handle_v1::ExtForeignToplevelHandleV1, Default::default())
127    ]);
128}

130impl<D> Dispatch<ext_foreign_toplevel_handle_v1::ExtForeignToplevelHandleV1, ForeignToplevelData, D>
131    for ForeignToplevelList
132where
133    D: Dispatch<ext_foreign_toplevel_handle_v1::ExtForeignToplevelHandleV1, ForeignToplevelData>
134        + ForeignToplevelListHandler,
135{
136    fn event(
137        state: &mut D,
138        handle: &ext_foreign_toplevel_handle_v1::ExtForeignToplevelHandleV1,
139        event: ext_foreign_toplevel_handle_v1::Event,
140        data: &ForeignToplevelData,
141        conn: &Connection,
142        qh: &QueueHandle<D>,
143    ) {
144        match event {
145            ext_foreign_toplevel_handle_v1::Event::Closed => {
146                state.toplevel_closed(conn, qh, handle.clone());
147                let toplevels = &mut state.foreign_toplevel_list_state().toplevels;
148                if let Some(idx) = toplevels.iter().position(|x| x == handle) {
149                    toplevels.remove(idx);
150                }
151                handle.destroy();
152            }
153            ext_foreign_toplevel_handle_v1::Event::Done => {
154                let mut inner = data.0.lock().unwrap();
155                let just_created = inner.current_info.is_none();
156                inner.current_info = Some(inner.pending_info.clone());
157                drop(inner);
158                if just_created {
159                    state.foreign_toplevel_list_state().toplevels.push(handle.clone());
160                    state.new_toplevel(conn, qh, handle.clone());
161                } else {
162                    state.update_toplevel(conn, qh, handle.clone());
163                }
164            }
165            ext_foreign_toplevel_handle_v1::Event::Title { title } => {
166                data.0.lock().unwrap().pending_info.title = title;
167            }
168            ext_foreign_toplevel_handle_v1::Event::AppId { app_id } => {
169                data.0.lock().unwrap().pending_info.app_id = app_id;
170            }
171            ext_foreign_toplevel_handle_v1::Event::Identifier { identifier } => {
172                data.0.lock().unwrap().pending_info.identifier = identifier;
173            }
174            _ => unreachable!(),
175        }
176    }
177}
