5/// Writes to a single log output stream.
6///
7/// Boxed instances of `LogWriter` can be used as additional log targets
8/// (see [writers](crate::writers) for more details).
9pub trait LogWriter: Send + Sync + AsAny {
10    /// Writes out a log line.
11    ///
12    /// # Errors
13    ///
14    /// [`std::io::Error`]
15    fn write(&self, now: &mut DeferredNow, record: &Record) -> std::io::Result<()>;
16
17    /// Flushes any buffered records.
18    ///
19    /// # Errors
20    ///
21    /// [`std::io::Error`]
22    fn flush(&self) -> std::io::Result<()>;
23
24    /// Provides the maximum log level that is to be written.
25    fn max_log_level(&self) -> log::LevelFilter {
26        log::LevelFilter::Trace
27    }
28
29    /// Sets the format function.
30    ///
31    /// Defaults to [`default_format`](crate::default_format),
32    /// but can be changed with a call to
33    /// [`Logger::format_for_writer`](crate::Logger::format_for_writer).
34    ///
35    /// The default implementation is a no-op.
36    fn format(&mut self, format: FormatFunction) {
37        _ = format;
38    }
39
40    /// Cleanup open resources, if necessary.
41    fn shutdown(&self) {}
42
43    /// Re-open the current output, if meaningful.
44    ///
45    /// This method is called from
46    /// [`LoggerHandle::reopen_output`](crate::LoggerHandle::reopen_output)
47    /// for all registered additional writers.
48    ///
49    /// # Errors
50    ///
51    /// Depend on registered writers.
52    fn reopen_output(&self) -> Result<(), FlexiLoggerError> {
53        Ok(())
54    }
55
56    /// Rotate the current output, if meaningful.
57    ///
58    /// This method is called from
59    /// [`LoggerHandle::trigger_rotation`](crate::LoggerHandle::trigger_rotation)
60    /// for all registered additional writers.
61    ///
62    /// # Errors
63    ///
64    /// Depend on registered writers.
65    fn rotate(&self) -> Result<(), FlexiLoggerError> {
66        Ok(())
67    }
68
69    // Takes a vec with three patterns per line that represent the log line,
70    // compares the written log with the expected lines,
71    // and asserts that both are in sync.
72    //
73    // This function is not meant for productive code, only for tests.
74    #[doc(hidden)]
75    fn validate_logs(&self, _expected: &[(&'static str, &'static str, &'static str)]) {
76        unimplemented!("only useful for tests");
77    }
78}
