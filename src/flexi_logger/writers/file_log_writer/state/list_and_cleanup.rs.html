22pub(super) fn existing_log_files(
23    file_spec: &FileSpec,
24    use_rotation: bool,
25    infix_filter: &InfixFilter,
26    selector: &LogfileSelector,
27) -> Vec<PathBuf> {
28    let mut result = Vec::new();
29    let related_files = file_spec.read_dir_related_files();
30    if use_rotation {
31        if selector.with_plain_files {
32            result.append(&mut file_spec.filter_files(
33                &related_files,
34                infix_filter,
35                file_spec.get_suffix().as_deref(),
36            ));
37        }
38        if selector.with_compressed_files {
39            result.append(&mut file_spec.filter_files(&related_files, infix_filter, Some("gz")));
40        }
41        if selector.with_r_current {
42            result.append(&mut file_spec.filter_files(
43                &related_files,
44                &InfixFilter::Equls(super::CURRENT_INFIX.to_string()),
45                file_spec.get_suffix().as_deref(),
46            ));
47        }
48        if let Some(ref custom_current) = selector.with_configured_current {
49            result.append(&mut file_spec.filter_files(
50                &related_files,
51                &InfixFilter::Equls(custom_current.clone()),
52                file_spec.get_suffix().as_deref(),
53            ));
54        }
55    } else {
56        result.push(file_spec.as_pathbuf(None));
57    }
58    result
59}

61pub(super) fn remove_or_compress_too_old_logfiles(
62    o_cleanup_thread_handle: Option<&CleanupThreadHandle>,
63    cleanup_config: &Cleanup,
64    file_spec: &FileSpec,
65    infix_filter: &InfixFilter,
66    writes_direct: bool,
67) -> Result<(), std::io::Error> {
68    o_cleanup_thread_handle.as_ref().map_or_else(
69        || {
70            remove_or_compress_too_old_logfiles_impl(
71                cleanup_config,
72                file_spec,
73                infix_filter,
74                writes_direct,
75            )
76        },
77        |cleanup_thread_handle| {
78            cleanup_thread_handle
79                .sender
80                .send(MessageToCleanupThread::Act)
81                .ok();
82            Ok(())
83        },
84    )
85}

87pub(crate) fn remove_or_compress_too_old_logfiles_impl(
88    cleanup_config: &Cleanup,
89    file_spec: &FileSpec,
90    infix_filter: &InfixFilter,
91    writes_direct: bool,
92) -> Result<(), std::io::Error> {
93    let (mut log_limit, day_limit, compress_limit): (usize, usize, usize) = match *cleanup_config {
94        Cleanup::Never => {
95            return Ok(());
96        }
97        Cleanup::KeepLogFiles(log_limit) => (log_limit, 0, 0),
98
99        Cleanup::KeepForDays(day_limit) => (usize::MAX, day_limit, 0),
100
101        #[cfg(feature = "compress")]
102        Cleanup::KeepCompressedFiles(compress_limit) => (0, 0, compress_limit),
103        #[cfg(feature = "compress")]
104        Cleanup::KeepLogAndCompressedFiles(log_limit, compress_limit) => {
105            (log_limit, 0, compress_limit)
106        }
107    };
108
109    // we must not clean up the current output file
110    if writes_direct && log_limit == 0 {
111        log_limit = 1;
112    }
113
114    for (index, file) in list_of_log_and_compressed_files(file_spec, infix_filter)
115        .into_iter()
116        .enumerate()
117    {
118        if index >= log_limit + compress_limit {
119            // delete (log or log.gz)
120            std::fs::remove_file(file)?;
121        } else if index >= log_limit {
122            #[cfg(feature = "compress")]
123            {
124                // compress, if not yet compressed
125                if let Some(extension) = file.extension() {
126                    if extension != "gz" {
127                        let mut compressed_file = file.clone();
128                        match compressed_file.extension() {
129                            Some(oss) => {
130                                let mut oss_gz = oss.to_os_string();
131                                oss_gz.push(".gz");
132                                compressed_file.set_extension(oss_gz.as_os_str());
133                            }
134                            None => {
135                                compressed_file.set_extension("gz");
136                            }
137                        }
138
139                        let mut gz_encoder = flate2::write::GzEncoder::new(
140                            File::create(compressed_file)?,
141                            flate2::Compression::fast(),
142                        );
143                        let mut old_file = File::open(file.clone())?;
144                        std::io::copy(&mut old_file, &mut gz_encoder)?;
145                        gz_encoder.finish()?;
146                        std::fs::remove_file(&file)?;
147                    }
148                }
149            }
150        } else if day_limit > 0 {
151            // Remove files older than the configured day limit
152            let mod_limit = std::time::SystemTime::now()
153                .checked_sub(std::time::Duration::from_secs(
154                    u64::try_from(day_limit)
155                        .ok()
156                        .and_then(|days| days.checked_mul(24 * 3600))
157                        .unwrap_or(u64::MAX),
158                ))
159                .unwrap_or(std::time::UNIX_EPOCH);
160
161            if std::fs::metadata(&file)?.modified()? < mod_limit {
162                std::fs::remove_file(&file)?;
163            }
164        }
165    }
166
167    Ok(())
168}

170const CLEANER: &str = "flexi_logger-fs-cleanup";

172#[derive(Debug)]
173pub(super) struct CleanupThreadHandle {
174    sender: std::sync::mpsc::Sender<MessageToCleanupThread>,
175    join_handle: JoinHandle<()>,
176}
177
178enum MessageToCleanupThread {
179    Act,
180    Die,
181}
182impl CleanupThreadHandle {
183    pub(super) fn shutdown(self) {
184        self.sender.send(MessageToCleanupThread::Die).ok();
185        self.join_handle.join().ok();
186    }
187}
188
189pub(super) fn start_cleanup_thread(
190    cleanup: Cleanup,
191    file_spec: FileSpec,
192    infix_filter: &InfixFilter,
193    writes_direct: bool,
194) -> Result<CleanupThreadHandle, std::io::Error> {
195    let (sender, receiver) = std::sync::mpsc::channel();
196    let builder = ThreadBuilder::new().name(CLEANER.to_string());
197    #[cfg(not(feature = "dont_minimize_extra_stacks"))]
198    let builder = builder.stack_size(512 * 1024);
199    let infix_filter_cp = infix_filter.clone();
200    Ok(CleanupThreadHandle {
201        sender,
202        join_handle: builder.spawn(move || {
203            while let Ok(MessageToCleanupThread::Act) = receiver.recv() {
204                remove_or_compress_too_old_logfiles_impl(
205                    &cleanup,
206                    &file_spec,
207                    &infix_filter_cp,
208                    writes_direct,
209                )
210                .ok();
211            }
212        })?,
213    })
