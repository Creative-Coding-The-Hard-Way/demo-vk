6pub(super) fn infix_from_timestamp(
7    ts: &DateTime<Local>,
8    use_utc: bool,
9    fmt: &InfixFormat,
10) -> String {
11    if use_utc {
12        ts.naive_utc().format(fmt.format())
13    } else {
14        ts.format(fmt.format())
15    }
16    .to_string()
17}

19fn ts_infix_from_path(path: &Path, file_spec: &FileSpec) -> String {
20    let idx = file_spec
21        .as_pathbuf(Some("rXXXXX"))
22        .to_string_lossy()
23        .find("rXXXXX")
24        .unwrap();
25    String::from_utf8_lossy(&path.to_string_lossy().as_bytes()[idx..idx + 20]).to_string()
26}

28pub(crate) fn timestamp_from_ts_infix(
29    infix: &str,
30    fmt: &InfixFormat,
31) -> Result<DateTime<Local>, String> {
32    match NaiveDateTime::parse_from_str(infix, fmt.format()) {
33        Ok(dt1) => Local
34            .from_local_datetime(&dt1)
35            .earliest()
36            .ok_or("Can't determine local time from infix".to_string()),
37        Err(e) if e.kind() == ParseErrorKind::NotEnough => {
38            match NaiveDate::parse_from_str(infix, fmt.format()) {
39                Ok(d1) => {
40                    Local
41                        .from_local_datetime(&d1.and_hms_opt(10, 0, 0).unwrap(/*OK*/))
42                        .earliest()
43                        .ok_or("Can't determine local time from infix".to_string())
44                }
45                Err(e) => Err(format!("Broken: {e:?}")),
46            }
47        }
48        Err(e) => Err(format!("Broken: {e:?}")),
49    }
50}

52pub(super) fn creation_timestamp_of_currentfile(
53    config: &FileLogWriterConfig,
54    current_infix: &str,
55    rotate_rcurrent: bool,
56    o_date_for_rotated_file: Option<&DateTime<Local>>,
57    fmt: &InfixFormat,
58) -> Result<DateTime<Local>, std::io::Error> {
59    let current_path = config.file_spec.as_pathbuf(Some(current_infix));
60
61    if rotate_rcurrent {
62        let date_for_rotated_file = o_date_for_rotated_file
63            .copied()
64            .unwrap_or_else(|| get_creation_timestamp(&current_path));
65        let rotated_path = path_for_rotated_file_from_timestamp(
66            &config.file_spec,
67            config.use_utc,
68            &date_for_rotated_file,
69            fmt,
70        );
71
72        match std::fs::rename(current_path.clone(), rotated_path.clone()) {
73            Ok(()) => {}
74            Err(e) => {
75                if e.kind() != std::io::ErrorKind::NotFound {
76                    return Err(e);
77                }
78            }
79        }
80    }
81    Ok(get_creation_timestamp(&current_path))
82}

84// determine the timestamp to which we want to write (file needn't exist)
85pub(super) fn latest_timestamp_file(
86    config: &FileLogWriterConfig,
87    rotate: bool,
88    fmt: &InfixFormat,
89) -> DateTime<Local> {
90    if rotate {
91        Local::now()
92    } else {
93        // find all file paths that fit the pattern
94        config
95            .file_spec
96            .list_of_files(
97                &InfixFilter::Numbrs,
98                config.file_spec.get_suffix().as_deref(),
99            )
100            .into_iter()
101            // retrieve the infix
102            .map(|path| ts_infix_from_path(&path, &config.file_spec))
103            // parse infix as date, ignore all infixes where this fails
104            .filter_map(|infix| timestamp_from_ts_infix(&infix, fmt).ok())
105            // take the newest of these dates
106            .reduce(|acc, e| if acc > e { acc } else { e })
107            // if nothing is found, take Local::now()
108            .unwrap_or_else(Local::now)
109    }
110}

112fn path_for_rotated_file_from_timestamp(
113    file_spec: &FileSpec,
114    use_utc: bool,
115    timestamp_for_rotated_file: &DateTime<Local>,
116    fmt: &InfixFormat,
117) -> PathBuf {
118    let infix = file_spec.collision_free_infix_for_rotated_file(&infix_from_timestamp(
119        timestamp_for_rotated_file,
120        use_utc,
121        fmt,
122    ));
123    file_spec.as_pathbuf(Some(&infix))
124}
