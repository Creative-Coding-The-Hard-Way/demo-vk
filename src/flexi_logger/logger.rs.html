38/// The entry-point for using `flexi_logger`.
39///
40/// `Logger` is a builder class that allows you to
41/// * specify your desired (initial) loglevel-specification
42///   * either as a String ([`Logger::try_with_str`])
43///   * or by providing it in the environment ([`Logger::try_with_env`]),
44///   * or by combining both options ([`Logger::try_with_env_or_str`]),
45///   * or by building a [`LogSpecification`] programmatically ([`Logger::with`]),
46/// * use the desired configuration methods,
47/// * and finally start the logger with
48///
49///   * [`Logger::start`], or
50///   * [`Logger::start_with_specfile`].
51///
52/// # Usage
53///
54/// See [`code_examples`](code_examples/index.html) for a comprehensive list of usage possibilities.
55pub struct Logger {
56    spec: LogSpecification,
57    log_target: LogTarget,
58    duplicate_err: Duplicate,
59    duplicate_out: Duplicate,
60    format_for_file: FormatFunction,
61    format_for_stderr: FormatFunction,
62    format_for_stdout: FormatFunction,
63    format_for_writer: FormatFunction,
64    #[cfg(feature = "colors")]
65    o_palette: Option<String>,
66    flush_interval: std::time::Duration,
67    flwb: FileLogWriterBuilder,
68    other_writers: HashMap<String, Box<dyn LogWriter>>,
69    filter: Option<Box<dyn LogLineFilter + Send + Sync>>,
70    error_channel: ErrorChannel,
71    use_utc: bool,
72    panic_on_error_channel_error: bool,
73}
74
75enum LogTarget {
76    StdErr,
77    StdOut,
78    // use_file_writer, optional additional writer
79    Multi(bool, Option<Box<dyn LogWriter>>),
80}

85    /// Creates a Logger that you provide with an explicit [`LogSpecification`].
86    ///
87    /// ## Examples
88    ///
89    /// ```rust
90    /// use log::LevelFilter;
91    /// use flexi_logger::Logger;
92    /// let logger = Logger::with(LevelFilter::Info).start().unwrap();
93    /// ```
94    ///
95    /// ```rust
96    /// use flexi_logger::{Logger, LogSpecification};
97    /// let logger = Logger::with(
98    ///         LogSpecification::parse("info, critical_mod = trace").unwrap()
99    ///     ).start().unwrap();
100    /// ```
101    #[must_use]
102    pub fn with(logspec: impl Into<LogSpecification>) -> Self {
103        Self::from_spec_and_errs(logspec.into())
104    }

106    /// Creates a Logger that reads the [`LogSpecification`] from a `String` or `&str`.
107    /// See [`LogSpecification`] for the syntax.
108    ///
109    /// # Errors
110    ///
111    /// `FlexiLoggerError::Parse` if the String uses an erroneous syntax.
112    pub fn try_with_str<S: AsRef<str>>(s: S) -> Result<Self, FlexiLoggerError> {
113        Ok(Self::from_spec_and_errs(LogSpecification::parse(
114            s.as_ref(),
115        )?))
116    }

120    ///

123    /// # Errors
124    ///
125    /// `FlexiLoggerError::Parse` if the value of `RUST_LOG` is malformed.

131    /// `RUST_LOG`, or derives it from the given `String`, if `RUST_LOG` is not set.

133    /// # Errors
134    ///
135    /// `FlexiLoggerError::Parse` if the chosen value is malformed.
136    pub fn try_with_env_or_str<S: AsRef<str>>(s: S) -> Result<Self, FlexiLoggerError> {
137        Ok(Self::from_spec_and_errs(LogSpecification::env_or_parse(s)?))

140    fn from_spec_and_errs(spec: LogSpecification) -> Self {
141        #[cfg(feature = "colors")]
142        #[cfg(windows)]
143        {
144            nu_ansi_term::enable_ansi_support().ok();
145        }
146
147        Self {
148            spec,
149            log_target: LogTarget::StdErr,
150            duplicate_err: Duplicate::None,
151            duplicate_out: Duplicate::None,
152            format_for_file: default_format,
153
154            #[cfg(feature = "colors")]
155            format_for_stdout: AdaptiveFormat::Default
156                .format_function(std::io::stdout().is_terminal()),
157            #[cfg(feature = "colors")]
158            format_for_stderr: AdaptiveFormat::Default
159                .format_function(std::io::stderr().is_terminal()),
160
161            #[cfg(not(feature = "colors"))]
162            format_for_stdout: default_format,
163            #[cfg(not(feature = "colors"))]
164            format_for_stderr: default_format,
165
166            format_for_writer: default_format,
167            #[cfg(feature = "colors")]
168            o_palette: None,
169            flush_interval: ZERO_DURATION,
170            flwb: FileLogWriter::builder(FileSpec::default()),
171            other_writers: HashMap::<String, Box<dyn LogWriter>>::new(),
172            filter: None,
173            error_channel: ErrorChannel::default(),
174            use_utc: false,
175            panic_on_error_channel_error: true,
176        }
177    }

422    /// Prevent indefinite growth of the log file by applying file rotation
423    /// and a clean-up strategy for older log files.
424    ///
425    /// By default, the log file is fixed while your program is running and will grow indefinitely.
426    /// With this option being used, when the log file reaches the specified criterion,
427    /// the file will be closed and a new file will be opened.
428    ///
429    /// Note that also the filename pattern changes:
430    ///
431    /// - by default, no timestamp is added to the filename if rotation is used
432    /// - the logs are always written to a file with infix `_rCURRENT`
433    /// - when the rotation criterion is fulfilled, it is closed and renamed to a file
434    ///   with another infix (see `Naming`),
435    ///   and then the logging continues again to the (fresh) file with infix `_rCURRENT`.
436    ///
437    /// Example:
438    ///
439    /// After some logging with your program `my_prog` and rotation with `Naming::Numbers`,
440    /// you will find files like
441    ///
442    /// ```text
443    /// my_prog_r00000.log
444    /// my_prog_r00001.log
445    /// my_prog_r00002.log
446    /// my_prog_rCURRENT.log
447    /// ```
448    ///
449    /// ## Parameters
450    ///
451    /// `criterion` defines *when* the log file should be rotated, based on its size or age.
452    /// See [`Criterion`] for details.
453    ///
454    /// `naming` defines the naming convention for the rotated log files.
455    /// See [`Naming`] for details.
456    ///
457    /// `cleanup` defines the strategy for dealing with older files.
458    /// See [`Cleanup`] for details.
459    #[must_use]
460    pub fn rotate(mut self, criterion: Criterion, naming: Naming, cleanup: Cleanup) -> Self {
461        self.flwb = self.flwb.rotate(criterion, naming, cleanup);
462        self
463    }

636    /// By default, or with false, the file would be truncated.
637    ///
638    /// This option only has an effect if logs are written to files,
639    /// and it will hardly make an effect if `suppress_timestamp()` is not used.
640    #[must_use]
641    pub fn o_append(mut self, append: bool) -> Self {
642        self.flwb = self.flwb.o_append(append);
643        self
644    }
645
646    /// If a String is specified, it will be used on unix systems to create in the current folder
647    /// a symbolic link with this name to the current log file.
648    ///
649    /// This option only has an effect on unix systems and if logs are written to files.
650    #[must_use]
651    pub fn o_create_symlink<P: Into<PathBuf>>(mut self, symlink: Option<P>) -> Self {
652        self.flwb = self.flwb.o_create_symlink(symlink);
653        self
654    }

657/// Finally, start logging, optionally with a spec-file.
658impl Logger {
659    /// Consumes the Logger object and initializes `flexi_logger`.
660    ///
661    /// **Keep the [`LoggerHandle`] alive up to the very end of your program!**
662    /// Dropping the [`LoggerHandle`] flushes and shuts down [`FileLogWriter`]s
663    /// and other [`LogWriter`]s, and then may prevent further logging!
664    /// This should happen immediately before the program terminates, but not earlier.
665    ///
666    /// Dropping the [`LoggerHandle`] is uncritical
667    /// only with [`Logger::log_to_stdout`] or [`Logger::log_to_stderr`].
668    ///
669    /// The [`LoggerHandle`] also allows updating the log specification programmatically,
670    /// e.g. to intensify logging for (buggy) parts of a (test) program, etc.
671    ///
672    /// # Example
673    ///
674    /// ```rust
675    /// use flexi_logger::{Logger,WriteMode, FileSpec};
676    /// fn main() -> Result<(), Box<dyn std::error::Error>> {
677    ///     let _logger = Logger::try_with_str("info")?
678    ///         .log_to_file(FileSpec::default())
679    ///         .write_mode(WriteMode::BufferAndFlush)
680    ///         .start()?;
681    ///
682    ///     // ... do all your work and join back all threads whose logs you want to see ...
683    ///
684    ///     Ok(())
685    /// }
686    /// ```
687    ///
688    /// # Errors
689    ///
690    /// Several variants of [`FlexiLoggerError`] can occur.
691    pub fn start(self) -> Result<LoggerHandle, FlexiLoggerError> {
692        let (boxed_logger, handle) = self.build()?;
693        log::set_boxed_logger(boxed_logger)?;
694        Ok(handle)
695    }
696
697    /// Builds a boxed logger and a `LoggerHandle` for it,
698    /// but does not initialize the global logger.
699    ///
700    /// The returned boxed logger implements the [`Log`](log::Log) trait
701    /// and can be installed manually or nested within another logger.
702    ///
703    /// **Keep the [`LoggerHandle`] alive up to the very end of your program!**
704    /// See [`Logger::start`] for more details.
705    ///
706    /// # Errors
707    ///
708    /// Several variants of [`FlexiLoggerError`] can occur.
709    pub fn build(mut self) -> Result<(Box<dyn log::Log>, LoggerHandle), FlexiLoggerError> {
710        if self.use_utc {
711            self.flwb = self.flwb.use_utc();
712        }
713        set_panic_on_error_channel_error(self.panic_on_error_channel_error);
714
715        #[cfg(feature = "colors")]
716        set_palette(self.o_palette.as_deref())?;
717
718        let a_primary_writer = Arc::new(match self.log_target {
719            LogTarget::StdOut => {
720                if let WriteMode::SupportCapture = self.flwb.get_write_mode() {
721                    PrimaryWriter::test(true, self.format_for_stdout)
722                } else {
723                    PrimaryWriter::stdout(self.format_for_stdout, self.flwb.get_write_mode())
724                }
725            }
726            LogTarget::StdErr => {
727                if let WriteMode::SupportCapture = self.flwb.get_write_mode() {
728                    PrimaryWriter::test(false, self.format_for_stderr)
729                } else {
730                    PrimaryWriter::stderr(self.format_for_stderr, self.flwb.get_write_mode())
731                }
732            }
733            LogTarget::Multi(use_file_writer, mut o_writer) => PrimaryWriter::multi(
734                self.duplicate_err,
735                self.duplicate_out,
736                WriteMode::SupportCapture == *self.flwb.get_write_mode(),
737                self.format_for_stderr,
738                self.format_for_stdout,
739                if use_file_writer {
740                    Some(Box::new(
741                        self.flwb.format(self.format_for_file).try_build()?,
742                    ))
743                } else {
744                    None
745                },
746                {
747                    if let Some(ref mut writer) = o_writer {
748                        writer.format(self.format_for_writer);
749                    }
750                    o_writer
751                },
752            ),
753        });
754
755        let a_other_writers = Arc::new(self.other_writers);
756
757        if self.flush_interval != ZERO_DURATION {
758            start_flusher_thread(
759                Arc::clone(&a_primary_writer),
760                Arc::clone(&a_other_writers),
761                self.flush_interval,
762            )?;
763        }
764
765        let max_level = self.spec.max_level();
766        let a_l_spec = Arc::new(RwLock::new(self.spec));
767        set_error_channel(self.error_channel);
768
769        // initialize the lazy_statics in DeferredNow before threads are spawned
770        if self.use_utc {
771            DeferredNow::force_utc();
772        }
773        let mut now = DeferredNow::new();
774        now.now();
775
776        let flexi_logger = FlexiLogger::new(
777            Arc::clone(&a_l_spec),
778            Arc::clone(&a_primary_writer),
779            Arc::clone(&a_other_writers),
780            self.filter,
781        );
782
783        let handle = LoggerHandle::new(a_l_spec, a_primary_writer, a_other_writers);
784        handle.reconfigure(max_level);
785        Ok((Box::new(flexi_logger), handle))
786    }
787
788    /// Consumes the Logger object and initializes `flexi_logger` in a way that
789    /// subsequently the log specification can be updated,
790    /// while the program is running, by editing a file.
791    ///
792    /// Uses the spec that was given to the factory method ([`Logger::with`] etc)
793    /// as initial spec and then tries to read the logspec from a file.
794    ///
795    /// If the file does not exist, `flexi_logger` creates the file and fills it
796    /// with the initial spec (and in the respective file format, of course).
797    ///
798    /// **Keep the returned [`LoggerHandle`] alive up to the very end of your program!**
799    /// See [`Logger::start`] for more details.
800    ///
801    /// # Feature dependency
802    ///
803    /// The implementation of this configuration method uses some additional crates
804    /// that you might not want to depend on with your program if you don't use this functionality.
805    /// For that reason the method is only available if you activate the
806    /// `specfile` feature. See the usage section on
807    /// [crates.io](https://crates.io/crates/flexi_logger) for details.
808    ///
809    /// # Usage
810    ///
811    /// A logger initialization like
812    ///
813    /// ```rust,no_run
814    /// use flexi_logger::Logger;
815    /// Logger::try_with_str("info")
816    ///     .unwrap()
817    ///     // more logger configuration
818    ///     .start_with_specfile("logspecification.toml");
819    /// ```
820    ///
821    /// will create the file `logspecification.toml` (if it does not yet exist) with this content:
822    ///
823    /// ```toml
824    /// ### Optional: Default log level
825    /// global_level = 'info'
826    /// ### Optional: specify a regular expression to suppress all messages that don't match
827    /// #global_pattern = 'foo'
828    ///
829    /// ### Specific log levels per module are optionally defined in this section
830    /// [modules]
831    /// #'mod1' = 'warn'
832    /// #'mod2' = 'debug'
833    /// #'mod2::mod3' = 'trace'
834    /// ```
835    ///
836    /// You can subsequently edit and modify the file according to your needs,
837    /// while the program is running, and it will immediately take your changes into account.
838    ///
839    /// Currently only toml-files are supported, the file suffix thus must be `.toml`.
840    ///
841    /// The initial spec remains valid if the file cannot be read.
842    ///
843    /// If you update the specfile subsequently while the program is running, `flexi_logger`
844    /// re-reads it automatically and adapts its behavior according to the new content.
845    /// If the file cannot be read anymore, e.g. because the format is not correct, the
846    /// previous logspec remains active.
847    /// If the file is corrected subsequently, the log spec update will work again.
848    ///
849    /// # Errors
850    ///
851    /// Several variants of [`FlexiLoggerError`] can occur.
852    #[cfg_attr(docsrs, doc(cfg(feature = "specfile")))]
853    #[cfg(feature = "specfile_without_notification")]
854    pub fn start_with_specfile<P: AsRef<Path>>(
855        self,
856        specfile: P,
857    ) -> Result<LoggerHandle, FlexiLoggerError> {
858        let (boxed_logger, handle) = self.build_with_specfile(specfile)?;
859        log::set_boxed_logger(boxed_logger)?;
860        Ok(handle)
861    }
862
863    /// Builds a boxed logger and a `LoggerHandle` for it,
864    /// but does not initialize the global logger.
865    ///
866    /// See also [`Logger::start`] and [`Logger::start_with_specfile`].
867    /// for the properties of the returned logger.
868    ///
869    /// # Errors
870    ///
871    /// Several variants of [`FlexiLoggerError`] can occur.
872    #[cfg_attr(docsrs, doc(cfg(feature = "specfile")))]
873    #[cfg(feature = "specfile_without_notification")]
874    pub fn build_with_specfile<P: AsRef<Path>>(
875        self,
876        specfile: P,
877    ) -> Result<(Box<dyn log::Log>, LoggerHandle), FlexiLoggerError> {
878        let (boxed_log, mut handle) = self.build()?;
879
880        let specfile = specfile.as_ref();
881        synchronize_subscriber_with_specfile(&mut handle.writers_handle, specfile)?;
882
883        #[cfg(feature = "specfile")]
884        {
885            handle.oam_specfile_watcher = Some(Arc::new(Mutex::new(create_specfile_watcher(
886                specfile,
887                handle.writers_handle.clone(),
888            )?)));
889        }
890
891        Ok((boxed_log, handle))
892    }
893}

895// Reread the specfile when it was updated
896#[cfg(feature = "specfile")]
897pub(crate) fn create_specfile_watcher<S: LogSpecSubscriber>(
898    specfile: &Path,
899    mut subscriber: S,
900) -> Result<Debouncer<RecommendedWatcher>, FlexiLoggerError> {
901    let specfile = specfile
902        .canonicalize()
903        .map_err(FlexiLoggerError::SpecfileIo)?;
904    let clone = specfile.clone();
905    let parent = clone.parent().unwrap(/*cannot fail*/);
906
907    let mut debouncer = new_debouncer(
908        std::time::Duration::from_millis(1000),
909        move |res: DebounceEventResult| match res {
910            Ok(events) => events.iter().for_each(|e| {
911                if e.path
912                    .canonicalize()
913                    .map(|x| x == specfile)
914                    .unwrap_or(false)
915                {
916                    log_spec_string_from_file(&specfile)
917                        .map_err(FlexiLoggerError::SpecfileIo)
918                        .and_then(LogSpecification::from_toml)
919                        .and_then(|spec| subscriber.set_new_spec(spec))
920                        .map_err(|e| {
921                            eprint_err(
922                                ErrorCode::LogSpecFile,
923                                "continuing with previous log specification, because \
924                                            rereading the log specification file failed",
925                                &e,
926                            );
927                        })
928                        .ok();
929                }
930            }),
931            Err(e) => eprint_err(
932                ErrorCode::LogSpecFile,
933                "error while watching the specfile",
934                &e,
935            ),
936        },
937    )
938    .unwrap();
939
940    debouncer
941        .watcher()
942        .watch(parent, RecursiveMode::NonRecursive)
943        .unwrap();
944
945    Ok(debouncer)
946}

948// If the specfile exists, read the file and update the subscriber's logspec from it;
949// otherwise try to create the file, with the current spec as content, under the specified name.
950#[cfg(feature = "specfile_without_notification")]
951pub(crate) fn synchronize_subscriber_with_specfile<S: LogSpecSubscriber>(
952    subscriber: &mut S,
953    specfile: &Path,
954) -> Result<(), FlexiLoggerError> {
955    if specfile
956        .extension()
957        .unwrap_or_else(|| std::ffi::OsStr::new(""))
958        .to_str()
959        .unwrap_or("")
960        != "toml"
961    {
962        return Err(FlexiLoggerError::SpecfileExtension(
963            "only spec files with extension toml are supported",
964        ));
965    }
966
967    if Path::is_file(specfile) {
968        let s = log_spec_string_from_file(specfile).map_err(FlexiLoggerError::SpecfileIo)?;
969        subscriber.set_new_spec(LogSpecification::from_toml(s)?)?;
970    } else {
971        if let Some(specfolder) = specfile.parent() {
972            std::fs::DirBuilder::new()
973                .recursive(true)
974                .create(specfolder)
975                .map_err(FlexiLoggerError::SpecfileIo)?;
976        }
977        let mut file = std::fs::OpenOptions::new()
978            .write(true)
979            .create_new(true)
980            .open(specfile)
981            .map_err(FlexiLoggerError::SpecfileIo)?;
982
983        subscriber.initial_spec()?.to_toml(&mut file)?;
984    }
985    Ok(())
986}

988#[cfg(feature = "specfile_without_notification")]
989pub(crate) fn log_spec_string_from_file<P: AsRef<Path>>(
990    specfile: P,
991) -> Result<String, std::io::Error> {
992    let mut buf = String::new();
993    let mut file = std::fs::File::open(specfile)?;
994    file.read_to_string(&mut buf)?;
995    Ok(buf)
996}

998/// Used to control which messages are to be duplicated to stderr, when `log_to_file()` is used.
999#[derive(Debug, Clone, Copy)]
1000pub enum Duplicate {
1001    /// No messages are duplicated.
1002    None = 0,
1003    /// Only error messages are duplicated.
1004    Error = 1,
1005    /// Error and warn messages are duplicated.
1006    Warn = 2,
1007    /// Error, warn, and info messages are duplicated.
1008    Info = 3,
1009    /// Error, warn, info, and debug messages are duplicated.
1010    Debug = 4,
1011    /// All messages are duplicated.
1012    Trace = 5,
1013    /// All messages are duplicated.
1014    All = 6,
1015}
1016impl From<u8> for Duplicate {
1017    fn from(val: u8) -> Self {
1018        match val {
1019            0 => Duplicate::None,
1020            1 => Duplicate::Error,
1021            2 => Duplicate::Warn,
1022            3 => Duplicate::Info,
1023            4 => Duplicate::Debug,
1024            5 => Duplicate::Trace,
1025            6 => Duplicate::All,
1026            _ => unreachable!(),
1027        }
1028    }
1029}
1030
1031impl From<LevelFilter> for Duplicate {
1032    fn from(level: LevelFilter) -> Self {
1033        match level {
1034            LevelFilter::Off => Duplicate::None,
1035            LevelFilter::Error => Duplicate::Error,
1036            LevelFilter::Warn => Duplicate::Warn,
1037            LevelFilter::Info => Duplicate::Info,
1038            LevelFilter::Debug => Duplicate::Debug,
1039            LevelFilter::Trace => Duplicate::Trace,
1040        }
1041    }
1042}
1043impl From<Duplicate> for LevelFilter {
1044    fn from(level: Duplicate) -> Self {
1045        match level {
1046            Duplicate::None => LevelFilter::Off,
1047            Duplicate::Error => LevelFilter::Error,
1048            Duplicate::Warn => LevelFilter::Warn,
1049            Duplicate::Info => LevelFilter::Info,
1050            Duplicate::Debug => LevelFilter::Debug,
1051            Duplicate::Trace | Duplicate::All => LevelFilter::Trace,
1052        }
1053    }
