1//! Describes how to extend `flexi_logger` with additional log writers.
2//!
3//! The module also contains two ready-to-use log writers,
4//! one for writing to files ([`FileLogWriter`]), one for writing to the syslog ([`SyslogWriter`]).
5//!
6//! Log writers can be used in two ways:
7//!
8//! * _Default output channel:_ <br>
9//!   You can influence to which output stream normal log messages will be written,
10//!   i.e. those from log macro calls without explicit target specification
11//!   (like in `log::error!("File not found")`).
12//!
13//!   With one of the methods
14//!
15//!   * [`Logger::log_to_stderr`](crate::Logger::log_to_stderr) (default)
16//!   * [`Logger::log_to_stdout`](crate::Logger::log_to_stdout)
17//!   * [`Logger::log_to_file`](crate::Logger::log_to_file)
18//!   * [`Logger::log_to_writer`](crate::Logger::log_to_writer)
19//!   * [`Logger::log_to_file_and_writer`](crate::Logger::log_to_file_and_writer)
20//!   * [`Logger::do_not_log`](crate::Logger::do_not_log)
21//!
22//!   you can change the default output channel. The fourth and the fifth of these methods
23//!   take log writers as input. See their documentation for more details.
24//!
25//!   Messages will only be written to the default output channel
26//!   if they match the current [log specification](crate::LogSpecification).
27//!
28//!   <br>
29//!
30//! * _Additional output channels:_ <br>
31//!   You can register additional log writers under a _target name_ with
32//!   [`Logger::add_writer()`](crate::Logger::add_writer), and address these log writers by
33//!   specifying the _target name_ in calls to the
34//!   [log macros](https://docs.rs/log/latest/log/macro.log.html).
35//!
36//!   The message of a log call with a _target value_ that has the form `{Name1,Name2,...}`, i.e.,
37//!   a comma-separated list of _target names_, within braces, is not sent to the default output
38//!   channel, but to the loggers specified explicitly in the list. In such a list
39//!   you can also specify the default output channel with the built-in target name `_Default`.
40//!
41//!   Log calls that are directed to an additional output channel will not be affected by
42//!   the value of `flexi_logger`'s log specification;
43//!   they will always be handed over to the respective `LogWriter`,
44//!   as you might want it for alerts or auditing.
45//!
46//!   In the following example we define an alert writer, and a macro to facilitate using it
47//!   (and avoid using the explicit target specification in the macro call), and
48//!   show some example calls.
49//!
50//!   ```rust
51//!   use log::*;
52//!
53//!   use flexi_logger::{FileSpec,Logger};
54//!   use flexi_logger::writers::FileLogWriter;
55//!
56//!   // Configure a FileLogWriter for alert messages
57//!   pub fn alert_logger() -> Box<FileLogWriter> {
58//!       Box::new(FileLogWriter::builder(
59//!           FileSpec::default()
60//!   #           .directory("log_files/writers_mod_docu")
61//!               .discriminant("Alert")
62//!               .suffix("alerts")
63//!           )
64//!           .print_message()
65//!           .try_build()
66//!           .unwrap())
67//!   }
68//!
69//!   // Define a macro for writing messages to the alert log and to the normal log
70//!   #[macro_use]
71//!   mod macros {
72//!       #[macro_export]
73//!       macro_rules! alert_error {
74//!           ($($arg:tt)*) => (
75//!               error!(target: "{Alert,_Default}", $($arg)*);
76//!           )
77//!       }
78//!   }
79//!
80//!   fn main() {
81//!       Logger::try_with_env_or_str("info")
82//!           .expect("LogSpecification String has errors")
83//!           .print_message()
84//!           .log_to_file(FileSpec::default())
85//!   #       .log_to_file(FileSpec::default().directory("log_files/writers_mod_docu"))
86//!           .add_writer("Alert", alert_logger())
87//!           .start()
88//!           .unwrap_or_else(|e| panic!("Logger initialization failed with {}", e));
89//!
90//!
91//!       // Explicitly send logs to different loggers
92//!       error!(target : "{Alert}", "This is only an alert");
93//!       error!(target : "{Alert,_Default}", "This is an alert and log message");
94//!
95//!       // Nicer: use the explicit macro
96//!       alert_error!("This is another alert and log message");
97//!
98//!       // Standard log macros write only to the normal log
99//!       error!("This is a normal error message");
100//!       warn!("This is a warning");
101//!       info!("This is an info message");
102//!       debug!("This is a debug message - you will not see it");
103//!       trace!("This is a trace message - you will not see it");
104//!   }
105//!
106//!   ```
107//!
