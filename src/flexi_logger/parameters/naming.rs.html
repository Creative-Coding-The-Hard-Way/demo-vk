1/// The naming convention for rotated log files.
2///
3/// Common rule for all variants is that the names of the current output file
4/// and the rotated log files only differ in the infix.
5///
6/// See [`Logger::log_to_file`](crate::Logger::log_to_file)
7/// for a description of how the filename is built, including the infix.
8///
9/// See the variants for how the infix is used by them.
10///
11/// Used in [`Logger::rotate`](crate::Logger::rotate).
12#[derive(Copy, Clone, Debug)]
13pub enum Naming {
14    /// Logs are written to a file with infix `rCURRENT`.
15    ///
16    /// File rotation renames this file to a name with a timestamp-infix
17    /// like `"r2023-01-27_14-41-08"`, logging continues with a fresh file with infix `rCURRENT`.
18    ///
19    /// If multiple rotations happen within the same second, extended infixes are used like
20    /// `"r2023-01-27_14-41-08.restart-0001"`.
21    ///
22    /// Same as
23    /// ```rust
24    /// # use flexi_logger::Naming;
25    /// # let dummy =
26    /// Naming::TimestampsCustomFormat {
27    ///    current_infix: Some("rCURRENT"),
28    ///     format: "r%Y-%m-%d_%H-%M-%S",
29    /// }
30    /// # ;
31    /// ```
32    Timestamps,
33
34    /// Logs are written to a file with a timestamp-infix, like `"r2023-01-27_14-41-08"`.
35    ///
36    /// File rotation switches over to the next file.
37    ///
38    /// If multiple rotations happen within the same second, extended infixes are used like
39    /// `"r2023-01-27_14-41-08.restart-0001"`.
40    ///
41    /// Same as
42    /// ```rust
43    /// # use flexi_logger::Naming;
44    /// # let dummy =
45    /// Naming::TimestampsCustomFormat {
46    ///    current_infix: None,
47    ///     format: "r%Y-%m-%d_%H-%M-%S",
48    /// }
49    /// # ;
50    /// ```
51    TimestampsDirect,
52
53    /// Defines the infixes for the file to which the logs are written, and for the rotated files.
54    TimestampsCustomFormat {
55        /// Controls if a special infix is used for the file to which the logs are currently
56        /// written.
57        ///
58        /// If `Some(infix)` is given, then it is taken as static infix for the file
59        /// to which the logs are written.
60        /// File rotation renames this file to a file with a timestamp infix.
61        /// If this file already exists, an extended infix is used like
62        /// `"2024-06-09.restart-0001"`.
63        ///
64        /// If `None` is given, then the logs will be directly written to a file with timestamp infix.
65        /// File rotation only switches over to a new file with a fresh timestamp infix.
66        /// If this file already exists, e.g. because rotation is triggered more frequently
67        /// than the timestamp varies (according to the pattern), then an extended infix is used like
68        /// `"2024-06-09.restart-0001"`.
69        current_infix: Option<&'static str>,
70        /// The format of the timestamp infix.
71        ///
72        /// See <https://docs.rs/chrono/latest/chrono/format/strftime/index.html> for a list of
73        /// supported specifiers.
74        ///
75        /// **Make sure to use a format**
76        ///
77        /// - that is compatible to your file system(s) (e.g., don't use slashes),
78        /// - that can be used by
79        ///   [chrono::NaiveDateTime](https://docs.rs/chrono/latest/chrono/naive/struct.NaiveDateTime.html#method.parse_from_str)
80        ///   or [chrono::NaiveDate](https://docs.rs/chrono/latest/chrono/naive/struct.NaiveDate.html#method.parse_from_str)
81        ///
82        /// Further, if you choose `current_infix` = `None` or `Some("")`, make sure to rotate only
83        /// by [age](crate::Criterion::Age), and choose an age that is not smaller than what
84        /// is expressed in the infix (e.g., don't rotate by minute if the infix only shows days).
85        ///
86        /// Examples:
87        ///
88        /// `"%Y-%m-%d"` produces timestamp infixes like `"2024-06-09"`.
89        ///
90        /// `"%Y-%m-%d_%H-%M-%S"` produces timestamp infixes like `"2024-06-09_13-24-35"`.
91        format: &'static str,
92    },
93
94    /// Logs are written to a file with infix `rCURRENT`.
95    ///
96    /// File rotation renames this file to a name with a number-infix
97    /// like `"r00000"`, `"r00001"`, etc.,
98    /// logging continues with a fresh file with infix `rCURRENT`.
99    Numbers,
100
101    /// Logs are written to a file with a number-infix,
102    /// like `"r00000"`, `"r00001"`, etc.
103    ///
104    /// File rotation switches over to the next file.
105    NumbersDirect,
106}
107impl Naming {
108    pub(crate) fn writes_direct(self) -> bool {
109        matches!(
110            self,
111            Naming::NumbersDirect
112                | Naming::TimestampsDirect
113                | Naming::TimestampsCustomFormat {
114                    current_infix: None | Some(""),
115                    format: _
116                }
117        )
118    }
119}
