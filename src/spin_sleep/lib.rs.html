1//! Accurate sleeping. Only use native sleep as far as it can be trusted, then spin.
2//!
3//! The problem with `thread::sleep` is it isn't always very accurate, and this accuracy varies
4//! on platform and state. Spinning is as accurate as we can get, but consumes the CPU
5//! rather ungracefully.
6//!
7//! This library adds a middle ground, using a configurable native accuracy setting allowing
8//! `thread::sleep` to wait the bulk of a sleep time, and spin the final section to guarantee
9//! accuracy.
10//!
11//! # Example: Replace `thread::sleep`
12//!
13//! The simplest usage with default native accuracy is a drop in replacement for `thread::sleep`.
14//! ```no_run
15//! # use std::time::Duration;
16//! spin_sleep::sleep(Duration::new(1, 12_550_000));
17//! ```
18//!
19//! # Example: Configure
20//! More advanced usage, including setting a custom native accuracy, can be achieved by
21//! constructing a `SpinSleeper`.
22//! ```no_run
23//! # use std::time::Duration;
24//! // Create a new sleeper that trusts native thread::sleep with 100μs accuracy
25//! let spin_sleeper = spin_sleep::SpinSleeper::new(100_000)
26//!     .with_spin_strategy(spin_sleep::SpinStrategy::YieldThread);
27//!
28//! // Sleep for 1.01255 seconds, this will:
29//! //  - thread:sleep for 1.01245 seconds, i.e., 100μs less than the requested duration
30//! //  - spin until total 1.01255 seconds have elapsed
31//! spin_sleeper.sleep(Duration::new(1, 12_550_000));
32//! ```
33//!
34//! Sleep can also be requested in `f64` seconds or `u64` nanoseconds
35//! (useful when used with `time` crate)
36//!
37//! ```no_run
38//! # use std::time::Duration;
39//! # let spin_sleeper = spin_sleep::SpinSleeper::new(100_000);
40//! spin_sleeper.sleep_s(1.01255);
41//! spin_sleeper.sleep_ns(1_012_550_000);
42//! ```
43//!
44//! OS-specific default settings should be good enough for most cases.
45//! ```
46//! # use spin_sleep::SpinSleeper;
47//! let sleeper = SpinSleeper::default();
48//! # let _ = sleeper;
49//! ```

70/// Accuracy container for spin sleeping. See [crate docs](index.html).
71#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
72pub struct SpinSleeper {
73    native_accuracy_ns: u32,
74    spin_strategy: SpinStrategy,
75}
76
77#[cfg(not(windows))]
78const DEFAULT_NATIVE_SLEEP_ACCURACY: SubsecondNanoseconds = 125_000;
79
80/// Asks the OS to put the current thread to sleep for at least the specified amount of time.
81/// **Does not spin.**
82///
83/// Equivalent to [`std::thread::sleep`], with the following exceptions:
84/// * **Windows** (>= Windows 10, version 1803): Uses a high resolution waitable timer, similar to std in rust >= 1.75.
85/// * **Windows** (< Windows 10, version 1803): Automatically selects the best native sleep accuracy
86///   generally achieving ~1ms native sleep accuracy, instead of default ~16ms.
87#[inline]
88pub fn native_sleep(duration: Duration) {
89    #[cfg(windows)]
90    windows::native_sleep(duration);
91
92    #[cfg(not(windows))]
93    thread::sleep(duration);
94}
95
96impl Default for SpinSleeper {
97    /// Constructs new SpinSleeper with defaults suiting the current OS
98    #[inline]
99    fn default() -> Self {
100        #[cfg(windows)]
101        let accuracy = windows::sleep_accuracy();
102        #[cfg(not(windows))]
103        let accuracy = DEFAULT_NATIVE_SLEEP_ACCURACY;
104
105        SpinSleeper::new(accuracy)
106    }
107}
108
109impl SpinSleeper {
110    /// Constructs new SpinSleeper with the input native sleep accuracy.
111    /// The lower the `native_accuracy_ns` the more we effectively trust the accuracy of the
112    /// [`native_sleep`] function.
113    #[inline]
114    pub fn new(native_accuracy_ns: SubsecondNanoseconds) -> SpinSleeper {
115        SpinSleeper {
116            native_accuracy_ns,
117            spin_strategy: <_>::default(),
118        }
119    }
120
121    /// Returns configured native_accuracy_ns.
122    pub fn native_accuracy_ns(self) -> SubsecondNanoseconds {
123        self.native_accuracy_ns
124    }
125
126    /// Returns configured spin strategy.
127    pub fn spin_strategy(self) -> SpinStrategy {
128        self.spin_strategy
129    }
130
131    /// Returns a spin sleeper with the given [`SpinStrategy`].
132    ///
133    /// # Example
134    /// ```
135    /// use spin_sleep::{SpinSleeper, SpinStrategy};
136    ///
137    /// let sleeper = SpinSleeper::default().with_spin_strategy(SpinStrategy::SpinLoopHint);
138    /// ```
139    pub fn with_spin_strategy(mut self, strategy: SpinStrategy) -> Self {
140        self.spin_strategy = strategy;
141        self
142    }
143
144    /// The internal `spin_sleep` method that puts the [current thread to sleep](fn.native_sleep.html)
145    /// for the duration less the configured native accuracy, then spins until the specified deadline.
146    #[inline]
147    fn spin_sleep(self, duration: Duration, deadline: Instant) {
148        let accuracy = Duration::new(0, self.native_accuracy_ns);
149        if duration > accuracy {
150            native_sleep(duration - accuracy);
151        }
152        // spin the rest of the duration
153        match self.spin_strategy {
154            SpinStrategy::YieldThread => {
155                while Instant::now() < deadline {
156                    thread::yield_now()
157                }
158            }
159            SpinStrategy::SpinLoopHint => {
160                while Instant::now() < deadline {
161                    std::hint::spin_loop()
162                }
163            }
164        }
165    }
166
167    /// Puts the [current thread to sleep](fn.native_sleep.html) for the `duration` less the
168    /// configured native accuracy. Then spins until the specified duration has elapsed.
169    pub fn sleep(self, duration: Duration) {
170        let deadline = Instant::now() + duration;
171        self.spin_sleep(duration, deadline);
172    }
173
174    /// Puts the [current thread to sleep](fn.native_sleep.html) until the `deadline` less
175    /// the configured native accuracy. Then spins until the specified deadline is reached.
176    pub fn sleep_until(self, deadline: Instant) {
177        let duration = deadline.saturating_duration_since(Instant::now());
178        self.spin_sleep(duration, deadline);
179    }
180
181    /// Puts the [current thread to sleep](fn.native_sleep.html) for the give seconds-duration
182    /// less the configured native accuracy. Then spins until the specified duration has elapsed.
183    pub fn sleep_s(self, seconds: Seconds) {
184        if seconds > 0.0 {
185            self.sleep(Duration::from_secs_f64(seconds));
186        }
187    }
188
189    /// Puts the [current thread to sleep](fn.native_sleep.html) for the give nanoseconds-duration
190    /// less the configured native accuracy. Then spins until the specified duration has elapsed.
191    pub fn sleep_ns(self, nanoseconds: Nanoseconds) {
192        self.sleep(Duration::from_nanos(nanoseconds))
193    }
194}
195
196/// Puts the [current thread to sleep](fn.native_sleep.html) for the `duration` less the
197/// default native accuracy. Then spins until the specified duration has elapsed.
198///
199/// Convenience function for `SpinSleeper::default().sleep(duration)`. Can directly take the
200/// place of `thread::sleep`.
201pub fn sleep(duration: Duration) {
202    SpinSleeper::default().sleep(duration);
203}
204
205/// Puts the [current thread to sleep](fn.native_sleep.html) until the `deadline` less
206/// the configured native accuracy. Then spins until the specified deadline is reached.
207///
208/// Convenience function for `SpinSleeper::default().sleep_until(instant)`. Can directly take
209/// the place of `thread::sleep_until`.
210pub fn sleep_until(deadline: Instant) {
211    SpinSleeper::default().sleep_until(deadline);
212}

214/// What to do while spinning.
215#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
216#[non_exhaustive]
217pub enum SpinStrategy {
218    /// Call [`std::thread::yield_now`] while spinning.
219    YieldThread,
220    /// Call [`std::hint::spin_loop`] while spinning.
221    SpinLoopHint,
222}
223
224/// Per-OS default strategy.
225/// * Windows  `SpinLoopHint`
226/// * !Windows `YieldThread`
227impl Default for SpinStrategy {
228    #[inline]
229    fn default() -> Self {
230        #[cfg(windows)]
231        return Self::SpinLoopHint;
232
233        #[cfg(not(windows))]
234        Self::YieldThread
235    }
236}

238// Not run unless specifically enabled with `cargo test --features "nondeterministic_tests"`
239// Travis does not do well with these tests, as they require a certain CPU priority.
240#[cfg(feature = "nondeterministic_tests")]
241#[cfg(test)]
242mod spin_sleep_test {
243    use super::*;
244
245    // The worst case error is unbounded even when spinning, but this accuracy is reasonable
246    // for most platforms.
247    const ACCEPTABLE_DELTA_NS: SubsecondNanoseconds = 50_000;
248
249    // Since on spin performance is not guaranteed it suffices that the assertions are valid
250    // 'most of the time'. This macro should avoid most 1-off failures.
251    macro_rules! passes_eventually {
252        ($test:expr) => {{
253            let mut error = None;
254            for _ in 0..50 {
255                match ::std::panic::catch_unwind(|| $test) {
256                    Ok(_) => break,
257                    Err(err) => {
258                        // test is failing, maybe due to spin unreliability
259                        error = error.or(Some(err));
260                        thread::sleep(Duration::new(0, 1000));
261                    }
262                }
263            }
264            assert!(
265                error.is_none(),
266                "Test failed 50/50 times: {:?}",
267                error.unwrap()
268            );
269        }};
270    }
271
272    #[test]
273    fn sleep_small() {
274        passes_eventually!({
275            let ns_duration = 12_345_678;
276
277            let ps = SpinSleeper::new(20_000_000);
278            ps.sleep(Duration::new(0, 1000)); // warm up
279
280            let before = Instant::now();
281            ps.sleep(Duration::new(0, ns_duration));
282            let elapsed = before.elapsed();
283
284            println!("Actual: {:?}", elapsed);
285            assert!(elapsed <= Duration::new(0, ns_duration + ACCEPTABLE_DELTA_NS));
286            assert!(elapsed >= Duration::new(0, ns_duration - ACCEPTABLE_DELTA_NS));
287        });
288    }
289
290    #[test]
291    fn sleep_big() {

309    fn sleep_s() {
310        passes_eventually!({
311            let ns_duration = 12_345_678_f64;

321            assert!(elapsed <= Duration::new(0, ns_duration.round() as u32 + ACCEPTABLE_DELTA_NS));
322            assert!(elapsed >= Duration::new(0, ns_duration.round() as u32 - ACCEPTABLE_DELTA_NS));
323        });
324    }
