7pub trait Encoded {
8    fn __as_bytes(&self) -> &[u8] {
9        self.__as_str().as_bytes()
10    }
11
12    fn __as_str(&self) -> &str;
13}

15#[derive(Clone, Debug)]
16pub struct EncodedChar {
17    buffer: [u8; MAX_UTF8_LENGTH],
18    length: usize,
19}
20
21impl Encoded for EncodedChar {
22    fn __as_str(&self) -> &str {
23        // SAFETY: This slice was encoded from a character.
24        unsafe { str::from_utf8_unchecked(&self.buffer[..self.length]) }
25    }
26}
27
28impl Encoded for &str {
29    fn __as_str(&self) -> &str {
30        self
31    }
32}

34/// Allows a type to be used for searching by [`RawOsStr`] and [`RawOsString`].
35///
36/// This trait is very similar to [`str::pattern::Pattern`], but its methods
37/// are private and it is implemented for different types.
38///
39/// [`RawOsStr`]: super::RawOsStr
40/// [`RawOsString`]: super::RawOsString
41#[cfg_attr(os_str_bytes_docs_rs, doc(cfg(feature = "raw_os_str")))]
42pub trait Pattern: private::Sealed {
43    #[doc(hidden)]
44    type __Encoded: Clone + Debug + Encoded;
45
46    #[doc(hidden)]
47    fn __encode(self) -> Self::__Encoded;
48}
49
50impl Pattern for char {
51    type __Encoded = EncodedChar;
52
53    fn __encode(self) -> Self::__Encoded {
54        let mut encoded = EncodedChar {
55            buffer: [0; MAX_UTF8_LENGTH],
56            length: 0,
57        };
58        encoded.length = self.encode_utf8(&mut encoded.buffer).len();
59        encoded
60    }
61}
62
63impl Pattern for &str {
64    type __Encoded = Self;
65
66    fn __encode(self) -> Self::__Encoded {
67        self
68    }
69}
70
71impl<'a> Pattern for &'a String {
72    type __Encoded = <&'a str as Pattern>::__Encoded;
73
74    fn __encode(self) -> Self::__Encoded {
75        (**self).__encode()
76    }
