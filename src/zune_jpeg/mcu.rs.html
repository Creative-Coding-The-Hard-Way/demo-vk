28/// The size of a DC block for a MCU.
29
30pub const DCT_BLOCK: usize = 64;

32impl<T: ZByteReaderTrait> JpegDecoder<T> {
33    /// Check for existence of DC and AC Huffman Tables
34    pub(crate) fn check_tables(&self) -> Result<(), DecodeErrors> {
35        // check that dc and AC tables exist outside the hot path
36        for component in &self.components {
37            let _ = &self
38                .dc_huffman_tables
39                .get(component.dc_huff_table)
40                .as_ref()
41                .ok_or_else(|| {
42                    DecodeErrors::HuffmanDecode(format!(
43                        "No Huffman DC table for component {:?} ",
44                        component.component_id
45                    ))
46                })?
47                .as_ref()
48                .ok_or_else(|| {
49                    DecodeErrors::HuffmanDecode(format!(
50                        "No DC table for component {:?}",
51                        component.component_id
52                    ))
53                })?;
54
55            let _ = &self
56                .ac_huffman_tables
57                .get(component.ac_huff_table)
58                .as_ref()
59                .ok_or_else(|| {
60                    DecodeErrors::HuffmanDecode(format!(
61                        "No Huffman AC table for component {:?} ",
62                        component.component_id
63                    ))
64                })?
65                .as_ref()
66                .ok_or_else(|| {
67                    DecodeErrors::HuffmanDecode(format!(
68                        "No AC table for component {:?}",
69                        component.component_id
70                    ))
71                })?;
72        }
73        Ok(())
74    }

76    /// Decode MCUs and carry out post processing.
77    ///
78    /// This is the main decoder loop for the library, the hot path.
79    ///
80    /// Because of this, we pull in some very crazy optimization tricks hence readability is a pinch
81    /// here.
82    #[allow(
83        clippy::similar_names,
84        clippy::too_many_lines,
85        clippy::cast_possible_truncation
86    )]
87    #[inline(never)]
88    pub(crate) fn decode_mcu_ycbcr_baseline(
89        &mut self, pixels: &mut [u8]
90    ) -> Result<(), DecodeErrors> {
91        setup_component_params(self)?;
92
93        // check dc and AC tables
94        self.check_tables()?;
95
96        let (mut mcu_width, mut mcu_height);
97
98        if self.is_interleaved {
99            // set upsampling functions
100            self.set_upsampling()?;
101
102            mcu_width = self.mcu_x;
103            mcu_height = self.mcu_y;
104        } else {
105            // For non-interleaved images( (1*1) subsampling)
106            // number of MCU's are the widths (+7 to account for paddings) divided bu 8.
107            mcu_width = ((self.info.width + 7) / 8) as usize;
108            mcu_height = ((self.info.height + 7) / 8) as usize;
109        }
110        if self.is_interleaved
111            && self.input_colorspace.num_components() > 1
112            && self.options.jpeg_get_out_colorspace().num_components() == 1
113            && (self.sub_sample_ratio == SampleRatios::V
114                || self.sub_sample_ratio == SampleRatios::HV)
115        {
116            // For a specific set of images, e.g interleaved,
117            // when converting from YcbCr to grayscale, we need to
118            // take into account mcu height since the MCU decoding needs to take
119            // it into account for padding purposes and the post processor
120            // parses two rows per mcu width.
121            //
122            // set coeff to be 2 to ensure that we increment two rows
123            // for every mcu processed also
124            mcu_height *= self.v_max;
125            mcu_height /= self.h_max;
126            self.coeff = 2;
127        }
128
129        if self.input_colorspace == ColorSpace::Luma && self.is_interleaved {
130            warn!("Grayscale image with down-sampled component, resetting component details");
131
132            self.reset_params();
133
134            mcu_width = ((self.info.width + 7) / 8) as usize;
135            mcu_height = ((self.info.height + 7) / 8) as usize;
136        }
137        let width = usize::from(self.info.width);
138
139        let padded_width = calculate_padded_width(width, self.sub_sample_ratio);
140
141        let mut stream = BitStream::new();
142        let mut tmp = [0_i32; DCT_BLOCK];
143
144        let comp_len = self.components.len();
145
146        for (pos, comp) in self.components.iter_mut().enumerate() {
147            // Allocate only needed components.
148            //
149            // For special colorspaces i.e YCCK and CMYK, just allocate all of the needed
150            // components.
151            if min(
152                self.options.jpeg_get_out_colorspace().num_components() - 1,
153                pos
154            ) == pos
155                || comp_len == 4
156            // Special colorspace
157            {
158                // allocate enough space to hold a whole MCU width
159                // this means we should take into account sampling ratios
160                // `*8` is because each MCU spans 8 widths.
161                let len = comp.width_stride * comp.vertical_sample * 8;
162
163                comp.needed = true;
164                comp.raw_coeff = vec![0; len];
165            } else {
166                comp.needed = false;
167            }
168        }
169
170        // If all components are contained in the first scan of MCUs, then we can process into
171        // (upsampled) pixels immediately after each MCU, for convenience we use each row of MCUS.
172        // Otherwise, we must first wait until following SOS provide the remaining components.
173        let all_components_in_first_scan = usize::from(self.num_scans) == self.components.len();
174        let mut progressive_mcus: [Vec<i16>; 4] = core::array::from_fn(|_| vec![]);
175
176        if !all_components_in_first_scan {
177            for (component, mcu) in self.components.iter().zip(&mut progressive_mcus) {
178                let len = mcu_width
179                    * component.vertical_sample
180                    * component.horizontal_sample
181                    * mcu_height
182                    * 64;
183                *mcu = vec![0; len];
184            }
185        }
186
187        let mut pixels_written = 0;
188
189        let is_hv = usize::from(self.is_interleaved);
190        let upsampler_scratch_size = is_hv * self.components[0].width_stride;
191        let mut upsampler_scratch_space = vec![0; upsampler_scratch_size];
192
193        'sos: loop {
194            trace!(
195                "Baseline decoding of components: {:?}",
196                &self.z_order[..usize::from(self.num_scans)]
197            );
198
199            for i in 0..mcu_height {
200                // Report if we have no more bytes
201                // This may generate false negatives since we over-read bytes
202                // hence that why 37 is chosen(we assume if we over-read more than 37 bytes, we have a problem)
203                if stream.overread_by > 37
204                // favourite number :)
205                {
206                    if self.options.strict_mode() {
207                        return Err(DecodeErrors::FormatStatic("Premature end of buffer"));
208                    };
209
210                    error!("Premature end of buffer");
211                    break;
212                }
213
214                // decode a whole MCU width,
215                // this takes into account interleaved components.
216                let terminate = if all_components_in_first_scan {
217                    self.decode_mcu_width::<false>(
218                        mcu_width,
219                        i,
220                        &mut tmp,
221                        &mut stream,
222                        &mut progressive_mcus
223                    )?
224                } else {
225                    /* NB: (cae). This code was added due to the issue at https://github.com/etemesi254/zune-image/issues/277
226                    *
227                    * There is a particular set of images that interleave the start of scan (SOS) with the MCU,
228                    * E.g if it's a three component image, we have SOS->MCU ->SOS->MCU ->SOS->MCU
229                    * which presents a problem on decoding, we need to buffer the whole image before continuing since
230                    * we won't have a row containing all the component data which will be needed e.g for color conversion.
231                    *
232                    * The mechanisms is that we decode the whole image upfront, which goes against the normal
233                    * routine of decoding MCU width , so this requires more memory upfront than initial routines
234                    * but it is a single image out of the many corpuses that exist, so its fine.
235                    * (image in test-images/jpeg/sos_news.jpeg)
236
237                    * Code contributed by  Aurelia Molzer (https://github.com/197g)
238
239                    *
240                    */
241
242                    self.decode_mcu_width::<true>(
243                        mcu_width,
244                        i,
245                        &mut tmp,
246                        &mut stream,
247                        &mut progressive_mcus
248                    )?
249                };
250
251                // process that width up until it's impossible. This is faster than allocation the
252                // full components, which we skipped earlier.
253                if all_components_in_first_scan {
254                    self.post_process(
255                        pixels,
256                        i,
257                        mcu_height,
258                        width,
259                        padded_width,
260                        &mut pixels_written,
261                        &mut upsampler_scratch_space
262                    )?;
263                }
264
265                match terminate {
266                    McuContinuation::Ok => {}
267                    McuContinuation::AnotherSos if all_components_in_first_scan => {
268                        warn!("More than one SOS despite already having all components");
269                        return Ok(());
270                    }
271                    McuContinuation::AnotherSos => continue 'sos,
272                    McuContinuation::Terminate => {
273                        warn!("Got terminate signal, will not process further");
274                        return Ok(());
275                    }
276                }
277            }
278
279            // Breaks if we get here, looping only if we have restarted, i.e. found another SOS and
280            // continued at `'sos'.
281            break;
282        }
283
284        if !all_components_in_first_scan {
285            self.finish_baseline_decoding(&progressive_mcus, mcu_width, pixels)?;
286        }
287
288        // it may happen that some images don't have the whole buffer
289        // so we can't panic in case of that
290        // assert_eq!(pixels_written, pixels.len());
291
292        // For UHD usecases that tie two images separating them with EOI and
293        // SOI markers, it may happen that we do not reach this image end of image
294        // So this ensures we reach it
295        // Ensure we read EOI
296        if !stream.seen_eoi {
297            let marker = get_marker(&mut self.stream, &mut stream);
298            match marker {
299                Ok(_m) => {
300                    trace!("Found marker {:?}", _m);
301                }
302                Err(_) => {
303                    // ignore error
304                }
305            }
306        }
307
308        trace!("Finished decoding image");
309
310        Ok(())
311    }

313    /// Process all MCUs when baseline decoding has been processing them component-after-component.
314    /// For simplicity this assembles the dequantized blocks in the order that the post processing
315    /// of an interleaved baseline decoding would use.
316    #[allow(clippy::too_many_lines)]
317    #[allow(clippy::cast_sign_loss)]
318    pub(crate) fn finish_baseline_decoding(
319        &mut self, block: &[Vec<i16>; MAX_COMPONENTS], _mcu_width: usize, pixels: &mut [u8]
320    ) -> Result<(), DecodeErrors> {
321        let mcu_height = self.mcu_y;
322
323        // Size of our output image(width*height)
324        let is_hv = usize::from(self.is_interleaved);
325        let upsampler_scratch_size = is_hv * self.components[0].width_stride;
326        let width = usize::from(self.info.width);
327        let padded_width = calculate_padded_width(width, self.sub_sample_ratio);
328
329        let mut upsampler_scratch_space = vec![0; upsampler_scratch_size];
330
331        for (pos, comp) in self.components.iter_mut().enumerate() {
332            // Mark only needed components for computing output colors.
333            if min(
334                self.options.jpeg_get_out_colorspace().num_components() - 1,
335                pos
336            ) == pos
337                || self.input_colorspace == ColorSpace::YCCK
338                || self.input_colorspace == ColorSpace::CMYK
339            {
340                comp.needed = true;
341            } else {
342                comp.needed = false;
343            }
344        }
345
346        let mut pixels_written = 0;
347
348        // dequantize and idct have been performed, only color convert.
349        for i in 0..mcu_height {
350            // All the data is already in the right order, we just need to be able to pass it to
351            // the post_process & upsample method. That expects all the data to be stored as one
352            // row of MCUs in each component's `raw_coeff`.
353            'component: for (position, component) in &mut self.components.iter_mut().enumerate() {
354                if !component.needed {
355                    continue 'component;
356                }
357
358                // step is the number of pixels this iteration wil be handling
359                // Given by the number of mcu's height and the length of the component block
360                // Since the component block contains the whole channel as raw pixels
361                // we this evenly divides the pixels into MCU blocks
362                //
363                // For interleaved images, this gives us the exact pixels comprising a whole MCU
364                // block
365                let step = block[position].len() / mcu_height;
366
367                // where we will be reading our pixels from.
368                let slice = &block[position][i * step..][..step];
369                let temp_channel = &mut component.raw_coeff;
370                temp_channel[..step].copy_from_slice(slice);
371            }
372
373            // process that whole stripe of MCUs
374            self.post_process(
375                pixels,
376                i,
377                mcu_height,
378                width,
379                padded_width,
380                &mut pixels_written,
381                &mut upsampler_scratch_space
382            )?;
383        }
384
385        return Ok(());
386    }

388    fn decode_mcu_width<const PROGRESSIVE: bool>(
389        &mut self, mcu_width: usize, mcu_height: usize, tmp: &mut [i32; 64],
390        stream: &mut BitStream, progressive: &mut [Vec<i16>; 4]
391    ) -> Result<McuContinuation, DecodeErrors> {
392        let z_order = self.z_order;
393
394        for j in 0..mcu_width {
395            // iterate over components
396            for &k in &z_order[..usize::from(self.num_scans)] {
397                let component = &mut self.components[k];
398
399                let dc_table = self.dc_huffman_tables[component.dc_huff_table % MAX_COMPONENTS]
400                    .as_ref()
401                    .unwrap();
402
403                let ac_table = self.ac_huffman_tables[component.ac_huff_table % MAX_COMPONENTS]
404                    .as_ref()
405                    .unwrap();
406
407                let qt_table = &component.quantization_table;
408                let channel = if PROGRESSIVE {
409                    let offset =
410                        mcu_height * component.width_stride * 8 * component.vertical_sample;
411                    &mut progressive[k][offset..]
412                } else {
413                    &mut component.raw_coeff
414                };
415
416                // If image is interleaved iterate over scan components,
417                // otherwise if it-s non-interleaved, these routines iterate in
418                // trivial scanline order(Y,Cb,Cr)
419                for v_samp in 0..component.vertical_sample {
420                    for h_samp in 0..component.horizontal_sample {
421                        // Fill the array with zeroes, decode_mcu_block expects
422                        // a zero based array.
423                        tmp.fill(0);
424
425                        let result = stream.decode_mcu_block(
426                            &mut self.stream,
427                            dc_table,
428                            ac_table,
429                            qt_table,
430                            tmp,
431                            &mut component.dc_pred
432                        );
433                        // If an error occurs we can either propagate it
434                        // as an error or print it and call terminate.
435                        //
436                        // This allows even corrupt images to render something,
437                        // even if its bad, matching browsers.
438                        //
439                        // See example in https://github.com/etemesi254/zune-image/issues/293
440                        if result.is_err() {
441                            return if self.options.strict_mode() {
442                                Err(result.err().unwrap())
443                            } else {
444                                error!("{}", result.err().unwrap());
445                                Ok(McuContinuation::Terminate)
446                            };
447                        }
448
449                        if component.needed {
450                            let idct_position = {
451                                // derived from stb and rewritten for my tastes
452                                let c2 = v_samp * 8;
453                                let c3 = ((j * component.horizontal_sample) + h_samp) * 8;
454
455                                component.width_stride * c2 + c3
456                            };
457
458                            let idct_pos = channel.get_mut(idct_position..).unwrap();
459                            //  call idct.
460                            (self.idct_func)(tmp, idct_pos, component.width_stride);
461                        }
462                    }
463                }
464            }
465
466            self.todo = self.todo.wrapping_sub(1);
467
468            if self.todo == 0 {
469                self.handle_rst_main(stream)?;
470                continue;
471            }
472
473            if stream.marker.is_some() && stream.bits_left == 0 {
474                break;
475            }
476        }
477
478        // After all interleaved components, that's an MCU
479        // handle stream markers
480        //
481        // In some corrupt images, it may occur that header markers occur in the stream.
482        // The spec EXPLICITLY FORBIDS this, specifically, in
483        // routine F.2.2.5  it says
484        // `The only valid marker which may occur within the Huffman coded data is the RSTm marker.`
485        //
486        // But libjpeg-turbo allows it because of some weird reason. so I'll also
487        // allow it because of some weird reason.
488        if let Some(m) = stream.marker {
489            if m == Marker::EOI {
490                // acknowledge and ignore EOI marker.
491                stream.marker.take();
492                trace!("Found EOI marker");
493                // Google Introduced the Ultra-HD image format which is basically
494                // stitching two images into one container.
495                // They basically separate two images via a EOI and SOI marker
496                // so let's just ensure if we ever see EOI, we never read past that
497                // ever.
498                // https://github.com/google/libultrahdr
499                stream.seen_eoi = true;
500            } else if let Marker::RST(_) = m {
501                //debug_assert_eq!(self.todo, 0);
502                if self.todo == 0 {
503                    self.handle_rst(stream)?;
504                }
505            } else if let Marker::SOS = m {
506                self.parse_marker_inner(m)?;
507                stream.marker.take();
508                stream.reset();
509                trace!("Found SOS marker");
510                return Ok(McuContinuation::AnotherSos);
511            } else {
512                if self.options.strict_mode() {
513                    return Err(DecodeErrors::Format(format!(
514                        "Marker {m:?} found where not expected"
515                    )));
516                }
517                error!(
518                    "Marker `{:?}` Found within Huffman Stream, possibly corrupt jpeg",
519                    m
520                );
521
522                self.parse_marker_inner(m)?;
523                stream.marker.take();
524                stream.reset();
525                return Ok(McuContinuation::Terminate);
526            }
527        }
528
529        Ok(McuContinuation::Ok)
530    }
531    // handle RST markers.
532    // No-op if not using restarts
533    // this routine is shared with mcu_prog
534    #[cold]
535    pub(crate) fn handle_rst(&mut self, stream: &mut BitStream) -> Result<(), DecodeErrors> {
536        self.todo = self.restart_interval;
537
538        if let Some(marker) = stream.marker {
539            // Found a marker
540            // Read stream and see what marker is stored there
541            match marker {
542                Marker::RST(_) => {
543                    // reset stream
544                    stream.reset();
545                    // Initialize dc predictions to zero for all components
546                    self.components.iter_mut().for_each(|x| x.dc_pred = 0);
547                    // Start iterating again. from position.
548                }
549                Marker::EOI => {
550                    // silent pass
551                }
552                _ => {
553                    return Err(DecodeErrors::MCUError(format!(
554                        "Marker {marker:?} found in bitstream, possibly corrupt jpeg"
555                    )));
556                }
557            }
558        }
559        Ok(())
560    }
561    #[allow(clippy::too_many_lines, clippy::too_many_arguments)]
562    pub(crate) fn post_process(
563        &mut self, pixels: &mut [u8], i: usize, mcu_height: usize, width: usize,
564        padded_width: usize, pixels_written: &mut usize, upsampler_scratch_space: &mut [i16]
565    ) -> Result<(), DecodeErrors> {
566        let out_colorspace_components = self.options.jpeg_get_out_colorspace().num_components();
567
568        let mut px = *pixels_written;
569        // indicates whether image is vertically up-sampled
570        let is_vertically_sampled = self
571            .components
572            .iter()
573            .any(|c| c.sample_ratio == SampleRatios::HV || c.sample_ratio == SampleRatios::V);
574
575        let mut comp_len = self.components.len();
576
577        // If we are moving from YCbCr -> Luma, we do not allocate storage for other components, so we
578        // will panic when we are trying to read samples, so for that case,
579        // hardcode it so that we  don't panic when doing
580        //   *samp = &samples[j][pos * padded_width..(pos + 1) * padded_width]
581        if out_colorspace_components < comp_len && self.options.jpeg_get_out_colorspace() == Luma {
582            comp_len = out_colorspace_components;
583        }
584        let mut color_conv_function =
585            |num_iters: usize, samples: [&[i16]; 4]| -> Result<(), DecodeErrors> {
586                for (pos, output) in pixels[px..]
587                    .chunks_exact_mut(width * out_colorspace_components)
588                    .take(num_iters)
589                    .enumerate()
590                {
591                    let mut raw_samples: [&[i16]; 4] = [&[], &[], &[], &[]];
592
593                    // iterate over each line, since color-convert needs only
594                    // one line
595                    for (j, samp) in raw_samples.iter_mut().enumerate().take(comp_len) {
596                        *samp = &samples[j][pos * padded_width..(pos + 1) * padded_width];
597                    }
598                    color_convert(
599                        &raw_samples,
600                        self.color_convert_16,
601                        self.input_colorspace,
602                        self.options.jpeg_get_out_colorspace(),
603                        output,
604                        width,
605                        padded_width
606                    )?;
607                    px += width * out_colorspace_components;
608                }
609                Ok(())
610            };
611
612        let comps = &mut self.components[..];
613
614        if self.is_interleaved && self.options.jpeg_get_out_colorspace() != ColorSpace::Luma {
615            for comp in comps.iter_mut() {
616                upsample(
617                    comp,
618                    mcu_height,
619                    i,
620                    upsampler_scratch_space,
621                    is_vertically_sampled
622                );
623            }
624
625            if is_vertically_sampled {
626                if i > 0 {
627                    // write the last line, it wasn't  up-sampled as we didn't have row_down
628                    // yet
629                    let mut samples: [&[i16]; 4] = [&[], &[], &[], &[]];
630
631                    for (samp, component) in samples.iter_mut().zip(comps.iter()) {
632                        *samp = &component.first_row_upsample_dest;
633                    }
634
635                    // ensure length matches for all samples
636                    let _first_len = samples[0].len();
637
638                    // This was a good check, but can be caused to panic, esp on invalid/corrupt images.
639                    // See one in issue https://github.com/etemesi254/zune-image/issues/262, so for now
640                    // we just ignore and generate invalid images at the end.
641
642                    //
643                    //
644                    // for samp in samples.iter().take(comp_len) {
645                    //     assert_eq!(first_len, samp.len());
646                    // }
647                    let num_iters = self.coeff * self.v_max;
648
649                    color_conv_function(num_iters, samples)?;
650                }
651
652                // After up-sampling the last row, save  any row that can be used for
653                // a later up-sampling,
654                //
655                // E.g the Y sample is not sampled but we haven't finished upsampling the last row of
656                // the previous mcu, since we don't have the down row, so save it
657                for component in comps.iter_mut() {
658                    if component.sample_ratio != SampleRatios::H {
659                        // We don't care about H sampling factors, since it's copied in the workers function
660
661                        // copy last row to be used for the  next color conversion
662                        let size = component.vertical_sample
663                            * component.width_stride
664                            * component.sample_ratio.sample();
665
666                        let last_bytes =
667                            component.raw_coeff.rchunks_exact_mut(size).next().unwrap();
668
669                        component
670                            .first_row_upsample_dest
671                            .copy_from_slice(last_bytes);
672                    }
673                }
674            }
675
676            let mut samples: [&[i16]; 4] = [&[], &[], &[], &[]];
677
678            for (samp, component) in samples.iter_mut().zip(comps.iter()) {
679                *samp = if component.sample_ratio == SampleRatios::None {
680                    &component.raw_coeff
681                } else {
682                    &component.upsample_dest
683                };
684            }
685
686            // we either do 7 or 8 MCU's depending on the state, this only applies to
687            // vertically sampled images
688            //
689            // for rows up until the last MCU, we do not upsample the last stride of the MCU
690            // which means that the number of iterations should take that into account is one less the
691            // up-sampled size
692            //
693            // For the last MCU, we upsample the last stride, meaning that if we hit the last MCU, we
694            // should sample full raw coeffs
695            let is_last_considered = is_vertically_sampled && (i != mcu_height.saturating_sub(1));
696
697            let num_iters = (8 - usize::from(is_last_considered)) * self.coeff * self.v_max;
698
699            color_conv_function(num_iters, samples)?;
700        } else {
701            let mut channels_ref: [&[i16]; MAX_COMPONENTS] = [&[]; MAX_COMPONENTS];
702
703            self.components
704                .iter()
705                .enumerate()
706                .for_each(|(pos, x)| channels_ref[pos] = &x.raw_coeff);
707
708            if let SampleRatios::Generic(_, v) = self.sub_sample_ratio {
709                color_conv_function(8 * v * self.coeff, channels_ref)?;
710            } else {
711                color_conv_function(8 * self.coeff, channels_ref)?;
712            }
713        }
714
715        *pixels_written = px;
716        Ok(())
717    }
718}

720enum McuContinuation {
721    Ok,
722    AnotherSos,
723    Terminate
