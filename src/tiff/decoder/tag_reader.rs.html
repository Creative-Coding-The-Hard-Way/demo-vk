9pub(crate) struct TagReader<'lt, R: EntryDecoder + ?Sized> {
10    pub(crate) decoder: &'lt mut R,
11    pub(crate) ifd: &'lt Directory,
12}
13
14pub(crate) trait EntryDecoder {
15    /// Turn an entry into a value by fetching the required bytes in the stream.
16    fn entry_val(&mut self, entry: &Entry) -> TiffResult<Value>;
17}
18
19impl<R: Seek + Read> EntryDecoder for ValueReader<R> {
20    fn entry_val(&mut self, entry: &Entry) -> TiffResult<Value> {
21        entry.val(&self.limits, self.bigtiff, &mut self.reader)
22    }
23}
24
25impl<'a, R: EntryDecoder + ?Sized> TagReader<'a, R> {
26    pub(crate) fn find_tag(&mut self, tag: Tag) -> TiffResult<Option<Value>> {
27        let entry = match self.ifd.get(tag) {
28            None => return Ok(None),
29            Some(entry) => entry.clone(),
30        };
31
32        self.decoder.entry_val(&entry).map(Some)
33    }
34
35    pub(crate) fn require_tag(&mut self, tag: Tag) -> TiffResult<Value> {
36        match self.find_tag(tag)? {
37            Some(val) => Ok(val),
38            None => Err(TiffError::FormatError(
39                TiffFormatError::RequiredTagNotFound(tag),
40            )),
41        }
42    }
43
44    pub fn find_tag_uint_vec<T: TryFrom<u64>>(&mut self, tag: Tag) -> TiffResult<Option<Vec<T>>> {
45        self.find_tag(tag)?
46            .map(|v| v.into_u64_vec())
47            .transpose()?
48            .map(|v| {
49                v.into_iter()
50                    .map(|u| {
51                        T::try_from(u).map_err(|_| TiffFormatError::InvalidTagValueType(tag).into())
52                    })
53                    .collect()
54            })
55            .transpose()
56    }
57}
