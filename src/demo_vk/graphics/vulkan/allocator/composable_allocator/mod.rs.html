24pub fn create_system_allocator(
25    logical_device: Arc<raii::Device>,
26    memory_properties: vk::PhysicalDeviceMemoryProperties,
27) -> impl ComposableAllocator {
28    let device_allocator = DeviceAllocator::new(logical_device.clone())
29        .description("DeviceAllocator", "The raw Vulkan device allocator.")
30        .shared();
31
32    FallbackAllocator::new(
33        DedicatedAllocator::new(device_allocator.clone()),
34        TypeIndexAllocator::new(move |index, addressable| {
35            // This is a silly implementation. By chaining split block
36            // allocators, a given allocation will continue to escalate until
37            // eventually asking the device for a big block of memory. From
38            // there, the block will be split into two parts over and over until
39            // the smallest correct block is found.
40            //
41            // This could certainly be made both *faster* (in CPU time) and
42            // more efficient in terms of allocated memory vs used memory. Not
43            // to mention fragmentation, etc...
44            //
45            // But this application doesn't care about that yet, so simply
46            // taking a big chunk and spliting it in half over and over is
47            // easy to understand and good enough for now.
48            #[rustfmt::skip]
49            let allocator =
50                SplitBlockAllocator::<_, 256                >::new(
51                SplitBlockAllocator::<_, 512                >::new(
52                SplitBlockAllocator::<_, 1024               >::new(
53                SplitBlockAllocator::<_, {1024 * 2         }>::new(
54                SplitBlockAllocator::<_, {1024 * 4         }>::new(
55                SplitBlockAllocator::<_, {1024 * 8         }>::new(
56                SplitBlockAllocator::<_, {1024 * 16        }>::new(
57                SplitBlockAllocator::<_, {1024 * 32        }>::new(
58                SplitBlockAllocator::<_, {1024 * 64        }>::new(
59                SplitBlockAllocator::<_, {1024 * 128       }>::new(
60                SplitBlockAllocator::<_, {1024 * 256       }>::new(
61                SplitBlockAllocator::<_, {1024 * 512       }>::new(
62                SplitBlockAllocator::<_, {1024 * 1024      }>::new(
63                SplitBlockAllocator::<_, {1024 * 1024 * 2  }>::new(
64                SplitBlockAllocator::<_, {1024 * 1024 * 4  }>::new(
65                SplitBlockAllocator::<_, {1024 * 1024 * 8  }>::new(
66                SplitBlockAllocator::<_, {1024 * 1024 * 16 }>::new(
67                SplitBlockAllocator::<_, {1024 * 1024 * 32 }>::new(
68                SplitBlockAllocator::<_, {1024 * 1024 * 64 }>::new(
69                SplitBlockAllocator::<_, {1024 * 1024 * 128}>::new(
70                    device_allocator.clone(),
71                ))))))))))))))))))));
72
73            allocator.description(
74                format!(
75                    "IndexedAllocator:{} - {}",
76                    index,
77                    if addressable {
78                        "DEVICE_ADDRESS"
79                    } else {
80                        "NOT ADDRESSED"
81                    }
82                ),
83                format!(
84                    "This allocator is responsible for allocating blocks \
85                    with memory properties, {:?}, and subdividing them for \
86                    use by the application.",
87                    memory_properties.memory_types[index as usize]
88                        .property_flags
89                ),
90            )
91        }),
92    )
93    .description(
94        "ApplicationAllocator",
95        "The top level allocator exposed to the application.",
96    )
97}

99/// Composable Allocators are externally synchronized and can have mutable
100/// state.
101pub trait ComposableAllocator {
102    /// Returns true if this allocator owns the given block.
103    fn owns(&self, block: &Block) -> bool;
104
105    /// Allocate a block of memory.
106    fn allocate_memory(
107        &mut self,
108        requirements: AllocationRequirements,
109    ) -> Result<Block>;
110
111    /// Free a block of memory.
112    fn free_memory(&mut self, block: &Block);
113
114    /// Replace self with a type-erased instance that can be cloned.
115    fn shared(self) -> SharedAllocator
116    where
117        Self: Sized + 'static,
118    {
119        SharedAllocator::new(self)
120    }
121}
122
123/// A reference counted shared instance of an allocator.
124#[derive(Clone)]
125pub struct SharedAllocator {
126    allocator: Rc<RefCell<dyn ComposableAllocator>>,
127}
128
129impl SharedAllocator {
130    pub fn new<A: ComposableAllocator + 'static>(allocator: A) -> Self {
131        Self {
132            allocator: Rc::new(RefCell::new(allocator)),
133        }
134    }
135}
136
137impl ComposableAllocator for SharedAllocator {
138    fn owns(&self, block: &Block) -> bool {
139        self.allocator.borrow().owns(block)
140    }
141
142    fn allocate_memory(
143        &mut self,
144        requirements: AllocationRequirements,
145    ) -> Result<Block> {
146        self.allocator.borrow_mut().allocate_memory(requirements)
147    }
148
149    fn free_memory(&mut self, block: &Block) {
150        self.allocator.borrow_mut().free_memory(block);
151    }
152}
