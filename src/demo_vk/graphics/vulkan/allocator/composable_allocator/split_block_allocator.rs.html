10/// Represents a Block of memory that has been split in half.
11struct SplitBlock {
12    first_in_use: bool,
13    second_in_use: bool,
14    first: Block,
15    second: Block,
16    total: Block,
17}
18
19impl SplitBlock {
20    pub fn new(block: Block) -> Result<Self> {
21        let midpoint = block.size() / 2;
22        Ok(Self {
23            first_in_use: false,
24            second_in_use: false,
25            first: block.subregion(0, midpoint)?,
26            second: block.subregion(midpoint, block.size() - midpoint)?,
27            total: block,
28        })
29    }
30
31    pub fn contains(&self, block: &Block) -> bool {
32        block.is_subregion_of(&self.total)
33    }
34
35    /// Attempts to take either the first or second free block. If no blocks are
36    /// available, then returns None.
37    pub fn take_free_block(&mut self) -> Option<Block> {
38        if !self.first_in_use {
39            self.first_in_use = true;
40            return Some(self.first);
41        }
42        if !self.second_in_use {
43            self.second_in_use = true;
44            return Some(self.second);
45        }
46        None
47    }
48
49    /// Returns the block so it can be reused.
50    pub fn free_block(&mut self, block: &Block) {
51        if self.first == *block {
52            self.first_in_use = false;
53        } else if self.second == *block {
54            self.second_in_use = false;
55        } else {
56            log::error!("Attempted to free an invalid block! {:#?}", block);
57        }
58    }
59
60    /// Returns true when the entire block is unused.
61    pub fn is_empty(&self) -> bool {
62        !self.first_in_use && !self.second_in_use
63    }
64}

66/// This allocator suballocates regions of a block by splitting it in two.
67///
68/// New blocks are provided by the composed allocator.
69pub struct SplitBlockAllocator<A: ComposableAllocator, const BLOCK_SIZE: u64> {
70    split_blocks: Vec<SplitBlock>,
71    allocator: A,
72}
73
74impl<A: ComposableAllocator, const BLOCK_SIZE: u64>
75    SplitBlockAllocator<A, BLOCK_SIZE>
76{
77    pub fn new(allocator: A) -> Self {
78        Self {
79            split_blocks: Vec::new(),
80            allocator,
81        }
82    }
83}
84
85impl<A: ComposableAllocator, const BLOCK_SIZE: u64> ComposableAllocator
86    for SplitBlockAllocator<A, BLOCK_SIZE>
87{
88    fn owns(&self, block: &Block) -> bool {
89        self.split_blocks.iter().any(|split| split.contains(block))
90    }
91
92    fn allocate_memory(
93        &mut self,
94        requirements: AllocationRequirements,
95    ) -> Result<Block> {
96        if requirements.allocation_size > BLOCK_SIZE
97            || !BLOCK_SIZE.is_multiple_of(requirements.alignment)
98        {
99            // Give up immediately if the block size is too small.
100            // Note: it's fine to give up if the block doesn't meet the
101            // alignment requirements. Eventually, this will reach the device
102            // allocator which is guaranteed to be maximally aligned for the
103            // platform.
104            return self.allocator.allocate_memory(requirements);
105        }
106
107        // check to see if any of the existing split blocks have space
108        let sub_block = self
109            .split_blocks
110            .iter_mut()
111            .find_map(|split_block| split_block.take_free_block());
112
113        if let Some(block) = sub_block {
114            return Ok(block);
115        }
116
117        // allocate a new block and split it instead
118        let block_requirements = AllocationRequirements {
119            alignment: BLOCK_SIZE,
120            allocation_size: BLOCK_SIZE * 2,
121            ..requirements
122        };
123        let mut new_split_block = SplitBlock::new(
124            self.allocator
125                .allocate_memory(block_requirements)
126                .with_context(trace!(
127                    "Unable to allocate a new block with requirements: {:#?}",
128                    block_requirements
129                ))?,
130        )?;
131
132        let block = new_split_block.take_free_block().unwrap();
133        self.split_blocks.push(new_split_block);
134
135        Ok(block)
136    }
137
138    fn free_memory(&mut self, block: &Block) {
139        let result = self
140            .split_blocks
141            .iter_mut()
142            .enumerate()
143            .find(|(_, split)| split.contains(block));
144
145        let free_index = if let Some((index, split_block)) = result {
146            split_block.free_block(block);
147            if split_block.is_empty() {
148                index
149            } else {
150                return;
151            }
152        } else {
153            // none of the split blocks own the given block, so send it up
154            self.allocator.free_memory(block);
155            return;
156        };
157        self.allocator
158            .free_memory(&self.split_blocks.swap_remove(free_index).total);
159    }
160}
