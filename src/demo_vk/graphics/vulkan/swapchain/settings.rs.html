11pub fn create_swapchain(
12    cxt: &VulkanContext,
13    framebuffer_size: (u32, u32),
14    previous_swapchain: Option<vk::SwapchainKHR>,
15) -> Result<(Arc<raii::Swapchain>, vk::Extent2D, vk::SurfaceFormatKHR)> {
16    let capabilities = unsafe {
17        cxt.surface_khr
18            .ext
19            .get_physical_device_surface_capabilities(
20                cxt.physical_device,
21                cxt.surface_khr.raw,
22            )
23            .with_context(trace!("Unable to get surface capabilities!"))?
24    };
25    log::trace!("Device capabilities:\n{:#?}", capabilities);
26
27    let format = select_image_format(cxt)?;
28    let extent = select_image_extent(&capabilities, framebuffer_size);
29    let queue_families = [cxt.graphics_queue_family_index];
30    let create_info = vk::SwapchainCreateInfoKHR {
31        surface: cxt.surface_khr.raw,
32        min_image_count: select_image_count(&capabilities),
33        image_format: format.format,
34        image_color_space: format.color_space,
35        image_extent: extent,
36        image_array_layers: 1,
37        image_usage: vk::ImageUsageFlags::COLOR_ATTACHMENT
38            | vk::ImageUsageFlags::TRANSFER_DST,
39        image_sharing_mode: vk::SharingMode::EXCLUSIVE,
40        queue_family_index_count: 1,
41        p_queue_family_indices: queue_families.as_ptr(),
42        pre_transform: capabilities.current_transform,
43        composite_alpha: vk::CompositeAlphaFlagsKHR::OPAQUE,
44        present_mode: select_present_mode(cxt)?,
45        clipped: vk::TRUE,
46        old_swapchain: previous_swapchain.unwrap_or(vk::SwapchainKHR::null()),
47        ..Default::default()
48    };
49    let swapchain = raii::Swapchain::new(cxt.device.clone(), &create_info)?;
50    log::trace!(
51        indoc::indoc!(
52            "
53            Created swapchain:
54              - swapchain: {:#?}
55              - extent: {:?}
56              - format: {:?}
57            "
58        ),
59        swapchain,
60        extent,
61        format
62    );
63    Ok((swapchain, extent, format))
64}

66/// Pick the desired image format for the swapchain.
67fn select_image_format(cxt: &VulkanContext) -> Result<vk::SurfaceFormatKHR> {
68    let surface_formats = unsafe {
69        cxt.surface_khr
70            .ext
71            .get_physical_device_surface_formats(
72                cxt.physical_device,
73                cxt.surface_khr.raw,
74            )
75            .with_context(trace!(
76                "Error while listing available surface formats!"
77            ))?
78    };
79    log::trace!("Formats supported by device\n{:#?}", surface_formats);
80
81    let preferred = surface_formats.iter().find(|surface_format| {
82        let has_color_space =
83            surface_format.color_space == vk::ColorSpaceKHR::SRGB_NONLINEAR;
84        let has_format = surface_format.format == vk::Format::B8G8R8A8_SRGB;
85        has_color_space && has_format
86    });
87
88    let format = preferred.or(surface_formats.first()).with_context(trace!(
89        "Unable to find a suitable surface format for the swapchain!"
90    ))?;
91
92    Ok(*format)
93}

95fn select_image_count(capabilities: &vk::SurfaceCapabilitiesKHR) -> u32 {
96    let count = capabilities.min_image_count + 2;
97    if capabilities.max_image_count > 0 {
98        count.clamp(capabilities.min_image_count, capabilities.max_image_count)
99    } else {
100        count
101    }
102}

104fn select_image_extent(
105    capabilities: &vk::SurfaceCapabilitiesKHR,
106    framebuffer_size: (u32, u32),
107) -> vk::Extent2D {
108    if capabilities.current_extent.width != u32::MAX {
109        // If current_extent does not equal u32::MAX, then it contains the
110        // extent of the targeted surface and can be used directly. See:
111        //
112        // https://registry.khronos.org/vulkan/specs/1.3-extensions/man/html/VkSurfaceCapabilitiesKHR.html
113
114        return capabilities.current_extent;
115    }
116
117    let (desired_width, desired_height) = framebuffer_size;
118    vk::Extent2D {
119        width: desired_width.clamp(
120            capabilities.min_image_extent.width,
121            capabilities.max_image_extent.width,
122        ),
123        height: desired_height.clamp(
124            capabilities.min_image_extent.height,
125            capabilities.max_image_extent.height,
126        ),
127    }
128}

130fn select_present_mode(cxt: &VulkanContext) -> Result<vk::PresentModeKHR> {
131    let present_modes = unsafe {
132        cxt.surface_khr
133            .ext
134            .get_physical_device_surface_present_modes(
135                cxt.physical_device,
136                cxt.surface_khr.raw,
137            )?
138    };
139    log::trace!("Present modes for device:\n{:#?}", present_modes);
140    if present_modes.contains(&vk::PresentModeKHR::MAILBOX) {
141        Ok(vk::PresentModeKHR::MAILBOX)
142    } else {
143        Ok(vk::PresentModeKHR::FIFO)
144    }
