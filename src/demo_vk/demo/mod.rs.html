49/// A demo is an opinionated application that automatically creates the
50/// VulkanContext, Swapchain, FramesInFlight and other common utilities.
51///
52/// The demo splits the application's update() function into two parts:
53/// * update(): update the application's state
54/// * draw(): build the frame's command buffer
55///
56/// Draw is separate because update() does not wait for the next swapchain
57/// image. There are some operations that may not depend on the swapchain and
58/// can reasonably be started before the frame's command buffer is ready to be
59/// recorded. (ticking a physics simulation, etc...)
60pub trait Demo {
61    type Args: Sized + Parser;
62    const FRAMES_IN_FLIGHT_COUNT: usize = 3;
63    const INITIAL_WINDOW_SIZE: (i32, i32) = (1024, 768);
64    const FRAMES_PER_SECOND: u32 = 120;
65
66    /// Creates a new instance of the demo.
67    /// The application is allowed to modify the window based on its own
68    /// requirements. This includes modifying the polling state, fullscreen
69    /// status, size, etc...
70    fn new(
71        window: &mut glfw::Window,
72        gfx: &mut Graphics<Self::Args>,
73    ) -> Result<Self>
74    where
75        Self: Sized;
76
77    /// Returns the required device features for this demo.
78    fn required_device_features() -> RequiredDeviceFeatures {
79        RequiredDeviceFeatures::default()
80    }
81
82    /// Handles a single GLFW event.
83    ///
84    /// This function is called in a loop to consume any pending events before
85    /// every call to update().
86    fn handle_event(
87        &mut self,
88        #[allow(unused_variables)] window: &mut glfw::Window,
89        #[allow(unused_variables)] gfx: &mut Graphics<Self::Args>,
90        #[allow(unused_variables)] event: glfw::WindowEvent,
91    ) -> Result<()> {
92        if let glfw::WindowEvent::Key(
93            glfw::Key::Escape,
94            _,
95            glfw::Action::Repeat,
96            _,
97        ) = event
98        {
99            window.set_should_close(true);
100        }
101        Ok(())
102    }
103
104    /// Called in a loop after all pending events have been processed.
105    ///
106    /// This is a good place for rendering logic. This method blocks event
107    /// processing, so it should be kept as responsive as possible.
108    fn update(
109        &mut self,
110        #[allow(unused_variables)] window: &mut glfw::Window,
111        #[allow(unused_variables)] gfx: &mut Graphics<Self::Args>,
112    ) -> Result<()> {
113        Ok(())
114    }
115
116    /// Build the command buffer for the next frame.
117    ///
118    /// Called after update() once the Frame is started.
119    fn draw(
120        &mut self,
121        #[allow(unused_variables)] window: &mut glfw::Window,
122        #[allow(unused_variables)] gfx: &mut Graphics<Self::Args>,
123        #[allow(unused_variables)] frame: &Frame,
124    ) -> Result<()> {
125        // By default, this method does nothing but transition the swapchain
126        // image to presentation format.
127        //
128        // Normally, a real application will use a render pass to do this.
129
130        let image_memory_barrier = vk::ImageMemoryBarrier {
131            old_layout: vk::ImageLayout::UNDEFINED,
132            new_layout: vk::ImageLayout::PRESENT_SRC_KHR,
133            image: frame.swapchain_image(),
134            subresource_range: vk::ImageSubresourceRange {
135                aspect_mask: vk::ImageAspectFlags::COLOR,
136                base_mip_level: 0,
137                level_count: 1,
138                base_array_layer: 0,
139                layer_count: 1,
140            },
141            ..Default::default()
142        };
143        unsafe {
144            gfx.vulkan.cmd_pipeline_barrier(
145                frame.command_buffer(),
146                vk::PipelineStageFlags::COLOR_ATTACHMENT_OUTPUT,
147                vk::PipelineStageFlags::COLOR_ATTACHMENT_OUTPUT,
148                vk::DependencyFlags::empty(),
149                &[],
150                &[],
151                &[image_memory_barrier],
152            );
153        }
154
155        Ok(())
156    }
157
158    /// Rebuild any of the demo's swapchain-dependent resources.
159    ///
160    /// All pending frames in flight are guaranteed to have completed by the
161    /// time this function is called. E.G. any previously-recorded Frame command
162    /// buffers are guaranteed to be finished.
163    fn rebuild_swapchain_resources(
164        &mut self,
165        #[allow(unused_variables)] window: &mut glfw::Window,
166        #[allow(unused_variables)] gfx: &mut Graphics<Self::Args>,
167    ) -> Result<()> {
168        Ok(())
169    }
170
171    /// Called when the application is paused.
172    ///
173    /// The application is paused automatically any time the framebuffer has
174    /// size of 0. This occurs when the application is minimized, etc..
175    fn paused(
176        &mut self,
177        #[allow(unused_variables)] window: &mut glfw::Window,
178        #[allow(unused_variables)] gfx: &mut Graphics<Self::Args>,
179    ) -> Result<()> {
180        Ok(())
181    }
182
183    /// Called when the application is unpaused.
184    fn unpaused(
185        &mut self,
186        #[allow(unused_variables)] window: &mut glfw::Window,
187        #[allow(unused_variables)] gfx: &mut Graphics<Self::Args>,
188    ) -> Result<()> {
189        Ok(())
190    }
191}

193struct DemoApp<D: Demo> {
194    graphics: Graphics<D::Args>,
195    demo: D,
196}
197
198impl<D: Demo> DemoApp<D> {
199    /// Called any time the framebuffer size may have changed.
200    /// Returns the current paused status for convenience.
201    fn framebuffer_size_changed(
202        &mut self,
203        window: &mut glfw::Window,
204    ) -> Result<bool> {
205        let (w, h) = window.get_framebuffer_size();
206        let should_pause = w == 0 || h == 0;
207
208        if should_pause {
209            if !self.graphics.paused {
210                self.demo.paused(window, &mut self.graphics)?;
211            }
212            self.graphics.paused = true;
213        } else {
214            if self.graphics.paused {
215                self.demo.unpaused(window, &mut self.graphics)?;
216            }
217            self.graphics.paused = false;
218            self.graphics.metrics.unpause();
219        }
220        Ok(self.graphics.paused)
221    }
222}
223
224impl<D: Demo + Sized> App for DemoApp<D> {
225    type Args = D::Args;
226
227    fn new(window: &mut glfw::Window, args: Self::Args) -> Result<Self>
228    where
229        Self: Sized,
230    {
231        window.set_size(D::INITIAL_WINDOW_SIZE.0, D::INITIAL_WINDOW_SIZE.1);
232        window.set_title(std::any::type_name::<D>());
233
234        let vulkan = VulkanContext::new(window, D::required_device_features())
235            .with_context(trace!("Unable to create the Vulkan Context!"))?;
236
237        let (w, h) = window.get_framebuffer_size();
238        let swapchain =
239            Swapchain::new(vulkan.clone(), (w as u32, h as u32), None)
240                .with_context(trace!("Unable to create the swapchain!"))?;
241
242        let frames_in_flight = FramesInFlight::new(
243            vulkan.clone(),
244            swapchain.images().len(),
245            D::FRAMES_IN_FLIGHT_COUNT,
246        )
247        .with_context(trace!("Unable to create frames in flight!"))?;
248
249        let fps_limiter = spin_sleep_util::interval(Duration::from_secs_f64(
250            1.0 / D::FRAMES_PER_SECOND as f64,
251        ));
252
253        let mut graphics = Graphics {
254            fps_limiter,
255            vulkan,
256            swapchain,
257            frames_in_flight,
258            args,
259            metrics: FrameMetrics::new(D::FRAMES_PER_SECOND as usize),
260            swapchain_needs_rebuild: false,
261            paused: false,
262        };
263        let demo = D::new(window, &mut graphics)
264            .with_context(trace!("Error initializing demo!"))?;
265
266        Ok(Self { graphics, demo })
267    }
268
269    fn handle_event(
270        &mut self,
271        window: &mut glfw::Window,
272        event: glfw::WindowEvent,
273    ) -> Result<()> {
274        self.demo.handle_event(window, &mut self.graphics, event)
275    }
276
277    fn update(&mut self, window: &mut glfw::Window) -> Result<()> {
278        if self.graphics.paused && self.framebuffer_size_changed(window)? {
279            std::hint::spin_loop();
280            return Ok(());
281        }
282
283        if self.graphics.swapchain_needs_rebuild {
284            self.graphics
285                .frames_in_flight
286                .wait_for_all_frames_to_complete()
287                .with_context(trace!(
288                    "Error waiting for frames before swapchain rebuild!"
289                ))?;
290            if self.framebuffer_size_changed(window)? {
291                return Ok(());
292            }
293            let (w, h) = window.get_framebuffer_size();
294            self.graphics.swapchain = Swapchain::new(
295                self.graphics.vulkan.clone(),
296                (w as u32, h as u32),
297                Some(self.graphics.swapchain.raw()),
298            )
299            .with_context(trace!("Error while rebuilding swapchain!"))?;
300
301            self.demo
302                .rebuild_swapchain_resources(window, &mut self.graphics)
303                .with_context(trace!(
304                    "Error while rebuilding demo swapchain resources!"
305                ))?;
306
307            self.graphics.swapchain_needs_rebuild = false;
308        }
309
310        self.graphics.metrics.frame_tick();
311
312        // Update application logic
313        // ------------------------
314
315        let before_update = Instant::now();
316        self.demo
317            .update(window, &mut self.graphics)
318            .with_context(trace!("Unhandled error in Demo::update()!"))?;
319        self.graphics.metrics.update_tick(before_update);
320
321        // Limit FPS, wait just before acquiring the frame
322        self.graphics.fps_limiter.tick();
323
324        // Prepare frame command buffer and submit
325        // ---------------------------------------
326
327        let before_draw = Instant::now();
328        let frame = match self
329            .graphics
330            .frames_in_flight
331            .start_frame(&self.graphics.swapchain)?
332        {
333            FrameStatus::FrameStarted(frame) => frame,
334            FrameStatus::SwapchainNeedsRebuild => {
335                self.graphics.swapchain_needs_rebuild = true;
336                return Ok(());
337            }
338        };
339
340        self.demo
341            .draw(window, &mut self.graphics, &frame)
342            .with_context(trace!("Unhandled error in Demo::draw()!"))?;
343
344        let result = self
345            .graphics
346            .frames_in_flight
347            .present_frame(&self.graphics.swapchain, frame)?;
348        if result == PresentImageStatus::SwapchainNeedsRebuild {
349            self.graphics.swapchain_needs_rebuild = true;
350        }
351        self.graphics.metrics.draw_tick(before_draw);
352
353        Ok(())
354    }
355}
