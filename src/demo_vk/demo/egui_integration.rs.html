22/// The EGUI painter is a complete EGUI integration for use within demos.
23///
24/// It maintains internal state for rendering directly to the swapchain image,
25/// and processing Winit events.
26pub struct EguiPainter {
27    state: egui_winit::State,
28    projection: Matrix4<f32>,
29    used_meshes: Vec<TrianglesMesh>,
30    free_meshes: Vec<TrianglesMesh>,
31    egui_textures: HashMap<egui::TextureId, i32>,
32    texture_loader: TextureLoader,
33    atlas: TextureAtlas,
34    renderer: StreamingRenderer,
35}

37/// Builds a projection matrix for the screen based on the current
38/// points_per_pixel and screen dimensions. This ensures that EGUI UI items
39/// correctly adhere to display scaling requirements for high-dpi displays.
40fn ortho_projection(
41    pixels_per_point: f32,
42    width: f32,
43    height: f32,
44) -> Matrix4<f32> {
45    let w = width / pixels_per_point;
46    let h = height / pixels_per_point;
47    #[rustfmt::skip]
48    let projection = Matrix4::new(
49        2.0 / w,  0.0,     0.0, -1.0,
50        0.0,     2.0 / h, 0.0, -1.0,
51        0.0,      0.0,     1.0, 0.0,
52        0.0,      0.0,     0.0, 1.0,
53    );
54    projection
55}

57/// Converts an EGUI clipping rect into a Vulkan vk::Rect2D, accounting for
58/// the pixels_per_point setting used by EGUI to scale UI elements.
59fn egui_rect_to_vk_rect(pixels_per_point: f32, rect: egui::Rect) -> vk::Rect2D {
60    vk::Rect2D {
61        offset: vk::Offset2D {
62            x: (pixels_per_point * rect.min.x) as i32,
63            y: (pixels_per_point * rect.min.y) as i32,
64        },
65        extent: vk::Extent2D {
66            width: (pixels_per_point * rect.width()) as u32,
67            height: (pixels_per_point * rect.height()) as u32,
68        },
69    }
70}

72impl EguiPainter {
73    pub fn new(graphics: &Graphics, window: &Window) -> Result<Self> {
74        let state = egui_winit::State::new(
75            egui::Context::default(),
76            egui::ViewportId::ROOT,
77            window,
78            Some(window.scale_factor() as f32),
79            None,
80            None,
81        );
82        let pixels_per_point =
83            egui_winit::pixels_per_point(state.egui_ctx(), window);
84        let PhysicalSize { width, height } = window.inner_size();
85        let atlas = unwrap_here!(
86            "Create EGUI painter texture atlas",
87            TextureAtlas::new(&graphics.vulkan)
88        );
89        let texture_loader = unwrap_here!(
90            "Create EGUI texture loader",
91            TextureLoader::new(graphics.vulkan.clone())
92        );
93        let renderer = unwrap_here!(
94            "Create EGUI painter streaming renderer instance",
95            StreamingRenderer::new(
96                &graphics.vulkan,
97                graphics.swapchain.format(),
98                &graphics.frames_in_flight,
99                &atlas
100            )
101        );
102        Ok(Self {
103            state,
104            projection: ortho_projection(
105                pixels_per_point,
106                width as f32,
107                height as f32,
108            ),
109            used_meshes: vec![],
110            free_meshes: vec![],
111            egui_textures: HashMap::new(),
112            texture_loader,
113            atlas,
114            renderer,
115        })
116    }

128    /// Draws the EGUI UI to the currently bound color attachment.
129    ///
130    /// # Safety
131    ///
132    /// - This function assumes that a render pass is already started (dynamic
133    ///   or not)
134    /// - This function assumes that the color attachment 0 is either a
135    ///   swapchain image or an image with the same dimensions and format.
136    /// - This function assumes that the viewport has already been set.
137    pub unsafe fn draw(&mut self, gfx: &Graphics, frame: &Frame) -> Result<()> {
138        self.renderer
139            .bind_texture_atlas(&gfx.vulkan, frame, &self.atlas);
140        unwrap_here!(
141            "Prepare EGUI clipped primitive meshes",
142            self.renderer.prepare_meshes(
143                &gfx.vulkan,
144                frame,
145                &self
146                    .used_meshes
147                    .iter()
148                    .map(|mesh| mesh as &dyn Mesh)
149                    .collect::<Vec<_>>(),
150            )
151        );
152        unwrap_here!(
153            "Write EGUI draw commands to frame command buffer",
154            self.renderer.write_draw_commands(&gfx.vulkan, frame)
155        );
156        Ok(())
157    }

159    pub fn run(
160        &mut self,
161        gfx: &Graphics,
162        window: &Window,
163        run_ui: impl FnMut(&egui::Context),
164    ) -> Result<()> {
165        let raw_input = {
166            let mut viewport_info = ViewportInfo::default();
167            egui_winit::update_viewport_info(
168                &mut viewport_info,
169                self.state.egui_ctx(),
170                window,
171                false,
172            );
173            let viewport_id = self.state.egui_input().viewport_id;
174            self.state
175                .egui_input_mut()
176                .viewports
177                .insert(viewport_id, viewport_info);
178
179            let PhysicalSize { width, height } = window.inner_size();
180            self.state.egui_input_mut().screen_rect = Some(egui::Rect {
181                min: egui::pos2(0.0, 0.0),
182                max: egui::pos2(width as f32, height as f32),
183            });
184
185            self.state.take_egui_input(window)
186        };
187        let full_output = self.state.egui_ctx().run(raw_input, run_ui);
188        self.state
189            .handle_platform_output(window, full_output.platform_output);
190
191        // create new textures if needed
192        for (egui_texture_id, delta) in &full_output.textures_delta.set {
193            let needs_replaced = if let Some(existing_texture_id) =
194                self.egui_textures.get(egui_texture_id)
195            {
196                // a texture exists, check to see if it needs to be reallocated
197                // based on the delta size
198                let existing_texture =
199                    self.atlas.get_texture(*existing_texture_id);
200
201                (existing_texture.width() as usize) < delta.image.width()
202                    || (existing_texture.height() as usize)
203                        < delta.image.height()
204            } else {
205                true // texture doesn't exist, so it always "needs replaced"
206            };
207
208            // create a new texture and update the texture map
209            if needs_replaced {
210                // no existing texture, so create a new one
211                log::info!("Creating egui texture: {:?}", egui_texture_id);
212
213                let texture = unwrap_here!(
214                    "Create EGUI texture",
215                    Texture::builder()
216                        .ctx(&gfx.vulkan)
217                        .dimensions((
218                            delta.image.width() as u32,
219                            delta.image.height() as u32,
220                        ))
221                        .format(vk::Format::R8G8B8A8_UNORM)
222                        .memory_property_flags(
223                            vk::MemoryPropertyFlags::DEVICE_LOCAL,
224                        )
225                        .image_usage_flags(
226                            vk::ImageUsageFlags::TRANSFER_DST
227                                | vk::ImageUsageFlags::SAMPLED,
228                        )
229                        .build()
230                );
231
232                self.egui_textures.insert(
233                    *egui_texture_id,
234                    self.atlas.add_texture(&gfx.vulkan, Arc::new(texture)),
235                );
236            }
237        }
238
239        // The previous block created any new textures that might be needed, so
240        // here we just need to write texture data.
241        for (egui_texture_id, delta) in full_output.textures_delta.set {
242            let texture_id = if let Some(texture_id) =
243                self.egui_textures.get(&egui_texture_id)
244            {
245                *texture_id
246            } else {
247                continue;
248            };
249            let texture = self.atlas.get_texture(texture_id);
250            let ImageData::Color(color) = delta.image;
251            let pixels = color
252                .pixels
253                .iter()
254                .flat_map(|color| color.to_srgba_unmultiplied())
255                .collect::<Vec<u8>>();
256            let offset = if let Some([w, h]) = delta.pos {
257                [w as u32, h as u32]
258            } else {
259                [0, 0]
260            };
261            let size = {
262                let [w, h] = color.size;
263                [w as u32, h as u32]
264            };
265            unwrap_here!(
266                "Update image data in EGUI texture",
267                self.texture_loader.tex_sub_image(
268                    &gfx.vulkan,
269                    texture,
270                    &pixels,
271                    offset,
272                    size,
273                )
274            );
275        }
276
277        let pixels_per_point = full_output.pixels_per_point;
278
279        // reset all meshes
280        self.free_meshes.extend(self.used_meshes.drain(0..));
281
282        // Fill mesh with geometry
283        let clipped_primitives = self.state.egui_ctx().tessellate(
284            full_output.shapes,
285            self.state.egui_ctx().pixels_per_point(),
286        );
287        let mut triangles_mesh = self.get_next_free_mesh();
288        let mut current_clip = vk::Rect2D::default();
289        if let Some(primitive) = clipped_primitives.first() {
290            current_clip =
291                egui_rect_to_vk_rect(pixels_per_point, primitive.clip_rect);
292            triangles_mesh.set_scissor(current_clip);
293        }
294
295        for clipped_primitive in clipped_primitives {
296            let clip_rect = egui_rect_to_vk_rect(
297                pixels_per_point,
298                clipped_primitive.clip_rect,
299            );
300            if clip_rect != current_clip {
301                // Allocate a new mesh and swap with the existing triangles_mesh
302                // so it can be saved in the 'used_meshes' list while the
303                // triangles_mesh remains the 'active' mesh for adding new
304                // vertices.
305                let mut next_mesh = self.get_next_free_mesh();
306                (triangles_mesh, next_mesh) = (next_mesh, triangles_mesh);
307                self.used_meshes.push(next_mesh);
308                triangles_mesh.set_scissor(clip_rect);
309            }
310
311            match clipped_primitive.primitive {
312                Primitive::Mesh(mesh) => {
313                    let texture_id = *self
314                        .egui_textures
315                        .get(&mesh.texture_id)
316                        .unwrap_or(&-1);
317                    triangles_mesh.indexed_triangles(
318                        mesh.vertices.iter().map(|egui_vertex| {
319                            Vertex::new(
320                                [egui_vertex.pos.x, egui_vertex.pos.y, 0.0],
321                                [egui_vertex.uv.x, egui_vertex.uv.y],
322                                egui_vertex.color.to_normalized_gamma_f32(),
323                                texture_id,
324                            )
325                        }),
326                        mesh.indices.iter().copied(),
327                    );
328                }
329                Primitive::Callback(_) => {
330                    log::warn!(
331                        "Callbacks unsupported by this backend (currently)"
332                    );
333                }
334            }
335        }
336        self.used_meshes.push(triangles_mesh);
337
338        Ok(())
339    }
340
341    fn get_next_free_mesh(&mut self) -> TrianglesMesh {
342        let mut mesh = self.free_meshes.pop().unwrap_or_else(|| {
343            TrianglesMesh::new(1_000, self.renderer.default_material().clone())
344        });
345        mesh.clear();
346        mesh.set_transform(self.projection);
347        mesh
348    }
