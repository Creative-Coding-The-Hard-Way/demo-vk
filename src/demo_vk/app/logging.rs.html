16// Used to remove verbose base dir prefix from log messages with file
17// information.
18static RELATIVE_ROOT_DIR: LazyLock<PathBuf> =
19    LazyLock::new(|| std::env::current_dir().unwrap());
20
21static LAST_NEWLINE_DELIM_MACHER: LazyLock<Regex> =
22    LazyLock::new(|| Regex::new(r"(│)(.*)$").unwrap());
23
24/// A global handle to the initialized flexi_logger.
25///
26/// This gets setup on the first call to setup_logger().
27static LOGGER_HANDLE: LazyLock<LoggerHandle> = LazyLock::new(|| {
28    Logger::try_with_env_or_str("trace")
29        .unwrap()
30        .log_to_file(FileSpec::default().directory("logs"))
31        .rotate(
32            Criterion::AgeOrSize(flexi_logger::Age::Hour, 1024 * 1024 * 8),
33            Naming::Timestamps,
34            flexi_logger::Cleanup::KeepLogFiles(3),
35        )
36        .format(multiline_format)
37        .duplicate_to_stdout(Duplicate::Debug)
38        .write_mode(WriteMode::Async)
39        .start()
40        .expect("Unable to start the logger!")
41});
42
43/// Initialize the multiline logger.
44pub fn setup() {
45    LOGGER_HANDLE.flush();
46}

48/// A multiline log format for flexi_logger.
49///
50/// Logs are automatically wrapped at terminal width and prefixed with unicode
51/// so it's easy to tell where a big log statement begins and ends.
52fn multiline_format(
53    w: &mut dyn std::io::Write,
54    now: &mut DeferredNow,
55    record: &Record,
56) -> Result<(), std::io::Error> {
57    let size = termwidth().min(80);
58    let wrap_options = Options::new(size)
59        .initial_indent("╭ ")
60        .subsequent_indent("│ ");
61
62    let file = Path::new(record.file().unwrap());
63    let file_message = if let Ok(relative_path) =
64        file.strip_prefix(RELATIVE_ROOT_DIR.as_path())
65    {
66        relative_path.to_str().unwrap().to_owned()
67    } else {
68        file.to_str().unwrap().to_owned()
69    };
70
71    let mut full_line = String::new();
72    writeln!(
73        full_line,
74        "{} [{}] [{}:{}]",
75        record.level(),
76        now.now().format("%H:%M:%S%.3f"),
77        file_message,
78        record.line().unwrap_or(0),
79    )
80    .expect("unable to format first log line");
81
82    write!(&mut full_line, "{}", &record.args())
83        .expect("unable to format log!");
84
85    let wrapped = textwrap::fill(&full_line, wrap_options);
86    let formatted = LAST_NEWLINE_DELIM_MACHER.replace(&wrapped, "╰$2");
87
88    writeln!(w, "{formatted}")
89}
